46  double precision function comp_source_time_function_gauss(t,hdur)
47 
48  use constants, only: pi
49 
50  implicit none
51 
52  double precision, intent(in) :: t,hdur
53  double precision :: hdur_decay,a
54 
55  ! note: hdur given is hdur_Gaussian = hdur/SOURCE_DECAY_MIMIC_TRIANGLE
56  ! and SOURCE_DECAY_MIMIC_TRIANGLE ~ 1.68
57  hdur_decay = hdur
58 
59  ! this here uses a stronger Gaussian decay rate (empirical value) to avoid non-zero onset times;
60  ! however, it should mimik a triangle source time function...
61  !hdur_decay = hdur / SOURCE_DECAY_STRONG
62 
63  ! note: a nonzero time to start the simulation with would lead to more high-frequency noise
64  ! due to the (spatial) discretization of the point source on the mesh
65 
66  ! Gaussian wavelet
67  a = 1.d0 / (hdur_decay**2)
68  comp_source_time_function_gauss = exp(-a * t**2) / (sqrt(pi)*hdur_decay)
69 
70  end function comp_source_time_function_gauss

337  double precision function marmousi_ormsby_wavelet(a)
338 
339  use constants, only: pi
340 
341  implicit none
342 
343  double precision :: sinc
344 
345  double precision, intent(in) :: a
346 
347  double precision :: f1,f2,f3,f4,b,c,tmp
348 
349  ! 5-10-60-80 Hz Ormsby Wavelet for Marmousi2 Model (Gray S. Martin, 2006)
350  ! Please find the Ormsby Wavelet here http://subsurfwiki.org/wiki/Ormsby_filter
351 
352  f1 = 5.0d0; ! low-cut frequency
353  f2 = 10.0d0; ! low-pass frequency
354  f3 = 60.0d0; ! high-pass frequency
355  f4 = 80.0d0; ! high-cut frequency
356 
357  b = sinc( f1 * a )
358  tmp = pi * f1
359  tmp = tmp * tmp
360  tmp = tmp / pi /( f2 - f1 )
361  c = b * b * tmp
362 
363  b = sinc( f2 * a )
364  tmp = pi * f2
365  tmp = tmp * tmp
366  tmp = tmp / pi /( f2 - f1 )
367  c = c - b * b * tmp
368 
369  b = sinc( f3 * a )
370  tmp = pi * f3
371  tmp = tmp * tmp
372  tmp = tmp / pi /( f4 - f3 )
373  c = c - b * b * tmp
374 
375  b = sinc( f4 * a )
376  tmp = pi * f4
377  tmp = tmp * tmp
378  tmp = tmp / pi /( f4 - f3 )
379  c = c + b * b * tmp
380 
381  marmousi_ormsby_wavelet = c
382 
383  end function marmousi_ormsby_wavelet
