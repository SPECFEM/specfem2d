34  subroutine compute_pressure_whole_medium()
35 
36 ! compute pressure in acoustic elements and in elastic elements
37 
38  use specfem_par, only: custom_real,ngllx,ngllz,nspec,ibool
39  use specfem_par_movie, only: vector_field_display
40 
41  implicit none
42 
43  ! local parameters
44  integer :: i,j,ispec,iglob
45  ! pressure in an element
46  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: pressure_element
47 
48  ! loop over spectral elements
49  do ispec = 1,nspec
50 
51  ! compute pressure in this element
52  call compute_pressure_one_element(ispec,pressure_element)
53 
54  ! use vector_field_display as temporary storage, store pressure in its second component
55  do j = 1,ngllz
56  do i = 1,ngllx
57  iglob = ibool(i,j,ispec)
58  vector_field_display(2,iglob) = pressure_element(i,j)
59  enddo
60  enddo
61 
62  enddo
63 
64  end subroutine compute_pressure_whole_medium

69 
70  subroutine compute_pressure_one_element(ispec,pressure_element)
71 
72 ! compute pressure in acoustic elements and in elastic elements
73 
74  use constants, only: custom_real,ngllx,ngllz,nglj,zero,two
75 
76  use specfem_par, only: n_sls,ispec_is_elastic,ispec_is_acoustic,ispec_is_poroelastic, &
77  ispec_is_gravitoacoustic,ispec_is_anisotropic,kmato,poroelastcoef,assign_external_model,vpext,vsext,rhoext, &
78  attenuation_viscoelastic,axisym,is_on_the_axis,displ_elastic,displs_poroelastic,displw_poroelastic, &
79  potential_dot_dot_acoustic,potential_acoustic,potential_dot_dot_gravitoacoustic,potential_gravitoacoustic, &
80  anisotropy,c11ext,c12ext,c13ext,c15ext,c23ext,c25ext,c33ext,c35ext,c55ext, &
81  hprimebar_xx,hprime_xx,hprime_zz,xix,xiz,gammax,gammaz,jacobian,ibool,coord,e1,e11,use_trick_for_better_pressure
82 
83  implicit none
84 
85  integer,intent(in) :: ispec
86  ! pressure in an element
87  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: pressure_element
88 
89  ! local variables
90  real(kind=CUSTOM_REAL) :: e1_sum,e11_sum
91  integer :: i_sls
92  integer :: i,j,k,iglob
93  real(kind=CUSTOM_REAL) :: sigma_yy !! ,sigmap
94  real(kind=CUSTOM_REAL) :: sigma_thetatheta
95  ! material properties of the elastic medium
96  real(kind=CUSTOM_REAL) :: denst
97  real(kind=CUSTOM_REAL) :: cpl,csl
98  real(kind=CUSTOM_REAL) :: mu_G,lambdal_G,lambdalplus2mul_G
99 
100  ! for anisotropy
101  double precision :: c11,c15,c13,c33,c35,c55,c12,c23,c25
102  ! Jacobian matrix and determinant
103  double precision :: xixl,xizl,gammaxl,gammazl
104  ! to evaluate cpI, cpII, and cs, and rI (poroelastic medium)
105  double precision :: phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar
106  double precision :: D_biot,H_biot,C_biot,M_biot
107  double precision :: mul_unrelaxed_elastic,lambdal_unrelaxed_elastic,lambdaplus2mu_unrelaxed_elastic
108  double precision :: sigma_xx,sigma_zz
109 
110  double precision :: dux_dxi,dux_dgamma,duz_dxi,duz_dgamma
111  double precision :: dwx_dxi,dwx_dgamma,dwz_dxi,dwz_dgamma
112  double precision :: dux_dxl,duz_dxl,dux_dzl,duz_dzl
113  double precision :: dwx_dxl,dwz_dzl
114 
115  double precision :: xxi
116 
117  ! initializes
118  pressure_element(:,:) = 0._custom_real
119 
120 ! if elastic element
121 !
122 ! from L. S. Bennethum, Compressibility Moduli for Porous Materials Incorporating Volume Fraction,
123 ! J. Engrg. Mech., vol. 132(11), p. 1205-1214 (2006), below equation (5):
124 ! for a 3D isotropic solid, pressure is defined in terms of the trace of the stress tensor as
125 ! p = -1/3 (t11 + t22 + t33) where t is the Cauchy stress tensor.
126 
127 ! to compute pressure in 3D in an elastic solid, one uses pressure = - trace(sigma) / 3
128 ! sigma_ij = lambda delta_ij trace(epsilon) + 2 mu epsilon_ij
129 ! = lambda (epsilon_xx + epsilon_yy + epsilon_zz) + 2 mu epsilon_ij
130 ! sigma_xx = lambda (epsilon_xx + epsilon_yy + epsilon_zz) + 2 mu epsilon_xx
131 ! sigma_yy = lambda (epsilon_xx + epsilon_yy + epsilon_zz) + 2 mu epsilon_yy
132 ! sigma_zz = lambda (epsilon_xx + epsilon_yy + epsilon_zz) + 2 mu epsilon_zz
133 ! pressure = - trace(sigma) / 3 = - (lambda + 2/3 mu) trace(epsilon) = - kappa * trace(epsilon)
134 !
135 ! to compute pressure in 2D in an elastic solid in the plane strain convention i.e. in the P-SV case,
136 ! one still uses pressure = - trace(sigma) / 3 but taking into account the fact
137 ! that the off-plane strain epsilon_zz is zero by definition of the plane strain convention
138 ! but thus the off-plane stress sigma_zz is not equal to zero,
139 ! one has instead: sigma_zz = lambda * (epsilon_xx + epsilon_yy), thus
140 ! sigma_ij = lambda delta_ij trace(epsilon) + 2 mu epsilon_ij
141 ! = lambda (epsilon_xx + epsilon_yy) + 2 mu epsilon_ij
142 ! sigma_xx = lambda (epsilon_xx + epsilon_yy) + 2 mu epsilon_xx
143 ! sigma_yy = lambda (epsilon_xx + epsilon_yy) + 2 mu epsilon_yy
144 ! sigma_zz = lambda * (epsilon_xx + epsilon_yy)
145 ! pressure = - trace(sigma) / 3 = - (lambda + 2*mu/3) (epsilon_xx + epsilon_yy)
146 
147  if (ispec_is_elastic(ispec)) then
148  ! elastic element
149 
150  ! get relaxed elastic parameters of current spectral element
151  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
152  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec))
153  lambdaplus2mu_unrelaxed_elastic = poroelastcoef(3,1,kmato(ispec))
154 
155  do j = 1,ngllz
156  do i = 1,ngllx
157 
158  !--- if external medium, get elastic parameters of current grid point
159  if (assign_external_model) then
160  cpl = vpext(i,j,ispec)
161  csl = vsext(i,j,ispec)
162  denst = rhoext(i,j,ispec)
163  mul_unrelaxed_elastic = denst*csl*csl
164  lambdal_unrelaxed_elastic = denst*cpl*cpl - two*mul_unrelaxed_elastic
165  endif
166 
167  ! derivative along x and along z
168  dux_dxi = zero
169  duz_dxi = zero
170 
171  dux_dgamma = zero
172  duz_dgamma = zero
173 
174  ! first double loop over GLL points to compute and store gradients
175  ! we can merge the two loops because NGLLX == NGLLZ
176 
177  if (axisym) then
178  if (is_on_the_axis(ispec)) then
179  do k = 1,nglj
180  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
181  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprimebar_xx(i,k)
182  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
183  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
184  enddo
185  else
186  do k = 1,nglj
187  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
188  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
189  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
190  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
191  enddo
192  endif
193  else
194  do k = 1,ngllx
195  dux_dxi = dux_dxi + displ_elastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
196  duz_dxi = duz_dxi + displ_elastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
197  dux_dgamma = dux_dgamma + displ_elastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
198  duz_dgamma = duz_dgamma + displ_elastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
199  enddo
200  endif
201 
202  xixl = xix(i,j,ispec)
203  xizl = xiz(i,j,ispec)
204  gammaxl = gammax(i,j,ispec)
205  gammazl = gammaz(i,j,ispec)
206 
207  ! derivatives of displacement
208  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
209  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
210 
211  if (axisym .and. is_on_the_axis(ispec) .and. i == 1) then ! d_uz/dr=0 on the axis
212  duz_dxl = 0.d0
213  endif
214 
215 ! compute diagonal components of the stress tensor (include attenuation or anisotropy if needed)
216 
217  if (attenuation_viscoelastic) then
218 
219 ! attenuation is implemented following the memory variable formulation of
220 ! J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
221 ! vol. 58(1), p. 110-120 (1993). More details can be found in
222 ! J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation in a linear
223 ! viscoelastic medium, Geophysical Journal International, vol. 95, p. 597-611 (1988).
224 
225 ! When implementing viscoelasticity according to the Carcione 1993 paper, attenuation is
226 ! non-causal rather than causal i.e. wave speed up instead of slowing down
227 ! when attenuation is turned on. We fixed that issue (which is not incorrect but non traditional)
228 ! by taking the unrelaxed state (infinite frequency) as a reference instead of the relaxed state (zero frequency)
229 ! and also using equations in Carcione's 2007 book.
230 ! See file doc/old_problem_attenuation_reference_Specfem2D_fixed_by_Xie_Zhinan.pdf
231 ! and doc/how_we_modified_Carcione_1993_to_make_it_causal_and_include_the_missing_1_over_L_factor.pdf
232 
233 ! See also J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
234 ! vol. 58(1), p. 110-120 (1993) for two memory-variable mechanisms (page 112).
235 
236 ! and J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation
237 ! in a linear viscoelastic medium, Geophysical Journal International,
238 ! vol. 95, p. 597-611 (1988) for two memory-variable mechanisms (page 604).
239 
240  if (axisym) then
241  if (is_on_the_axis(ispec)) then
242  if (is_on_the_axis(ispec) .and. i == 1) then ! First GLJ point
243  sigma_xx = 0._custom_real
244  sigma_zz = 0._custom_real
245  sigma_thetatheta = 0._custom_real
246  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
247  do k = 1,nglj
248  sigma_xx = sigma_xx + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
249  sigma_zz = sigma_zz + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
250  sigma_thetatheta = sigma_thetatheta + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
251  enddo
252  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl &
253  + lambdal_unrelaxed_elastic*sigma_xx/xxi
254  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
255  + lambdal_unrelaxed_elastic*sigma_zz/xxi
256  sigma_thetatheta = lambdal_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
257  + lambdaplus2mu_unrelaxed_elastic*sigma_thetatheta/xxi
258  else ! Not first GLJ point
259  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl &
260  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
261  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
262  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
263  sigma_thetatheta = lambdal_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
264  + lambdaplus2mu_unrelaxed_elastic &
265  * displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
266  endif
267  else ! Not on the axis
268  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl &
269  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
270  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
271  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
272  sigma_thetatheta = lambdal_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
273  + lambdaplus2mu_unrelaxed_elastic &
274  * displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
275  endif
276  sigma_yy = sigma_thetatheta
277  else ! Not axisym
278  ! compute the stress using the unrelaxed Lame parameters (Carcione 2007 page 125)
279  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl
280  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
281  sigma_yy = lambdal_unrelaxed_elastic*(dux_dxl + duz_dzl)
282  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl
283  endif
284 
285  ! add the memory variables using the relaxed parameters (Carcione 2007 page 125)
286  ! beware: there is a bug in Carcione's equation (2c) of his 1993 paper for sigma_zz, we fixed it in the code below
287  e1_sum = 0._custom_real
288  e11_sum = 0._custom_real
289 
290  do i_sls = 1,n_sls
291  e1_sum = e1_sum + e1(i,j,ispec,i_sls)
292  e11_sum = e11_sum + e11(i,j,ispec,i_sls)
293  enddo
294 
295  sigma_xx = sigma_xx + (lambdal_unrelaxed_elastic + mul_unrelaxed_elastic) * e1_sum &
296  + two * mul_unrelaxed_elastic * e11_sum
297  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
298  sigma_yy = sigma_yy + (lambdal_unrelaxed_elastic + mul_unrelaxed_elastic) * e1_sum
299  sigma_zz = sigma_zz + (lambdal_unrelaxed_elastic + mul_unrelaxed_elastic) * e1_sum &
300  - two * mul_unrelaxed_elastic * e11_sum
301 
302  else
303 
304  ! no attenuation
305 
306  if (axisym) then
307  if (is_on_the_axis(ispec)) then
308  if (is_on_the_axis(ispec) .and. i == 1) then ! First GLJ point
309  sigma_xx = 0._custom_real
310  sigma_zz = 0._custom_real
311  sigma_thetatheta = 0._custom_real
312  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
313  do k = 1,nglj
314  sigma_xx = sigma_xx + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
315  sigma_zz = sigma_zz + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
316  sigma_thetatheta = sigma_thetatheta + displ_elastic(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
317  enddo
318  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl &
319  + lambdal_unrelaxed_elastic*sigma_xx/xxi
320  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
321  + lambdal_unrelaxed_elastic*sigma_zz/xxi
322  sigma_thetatheta = lambdal_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
323  + lambdaplus2mu_unrelaxed_elastic*sigma_thetatheta/xxi
324  else ! Not first GLJ point
325  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl &
326  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
327  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
328  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
329  sigma_thetatheta = lambdal_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
330  + lambdaplus2mu_unrelaxed_elastic &
331  * displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
332  endif
333  else ! Not on the axis
334  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl &
335  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
336  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
337  + lambdal_unrelaxed_elastic*displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
338  sigma_thetatheta = lambdal_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl &
339  + lambdaplus2mu_unrelaxed_elastic &
340  * displ_elastic(1,ibool(i,j,ispec))/coord(1,ibool(i,j,ispec))
341  endif
342  sigma_yy = sigma_thetatheta
343  else ! Not axisym
344  sigma_xx = lambdaplus2mu_unrelaxed_elastic*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl
345  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
346  sigma_yy = lambdal_unrelaxed_elastic*(dux_dxl + duz_dzl)
347  sigma_zz = lambdaplus2mu_unrelaxed_elastic*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl
348  endif
349 
350  endif
351 
352  ! full anisotropy
353  if (ispec_is_anisotropic(ispec)) then
354  if (assign_external_model) then
355  c11 = c11ext(i,j,ispec)
356  c15 = c15ext(i,j,ispec)
357  c13 = c13ext(i,j,ispec)
358  c33 = c33ext(i,j,ispec)
359  c35 = c35ext(i,j,ispec)
360  c55 = c55ext(i,j,ispec)
361  c12 = c12ext(i,j,ispec)
362  c23 = c23ext(i,j,ispec)
363  c25 = c25ext(i,j,ispec)
364  else
365  c11 = anisotropy(1,kmato(ispec))
366  c13 = anisotropy(2,kmato(ispec))
367  c15 = anisotropy(3,kmato(ispec))
368  c33 = anisotropy(4,kmato(ispec))
369  c35 = anisotropy(5,kmato(ispec))
370  c55 = anisotropy(6,kmato(ispec))
371  c12 = anisotropy(7,kmato(ispec))
372  c23 = anisotropy(8,kmato(ispec))
373  c25 = anisotropy(9,kmato(ispec))
374  endif
375 
376  duz_dxl = duz_dxi*xixl + duz_dgamma*gammaxl
377  dux_dzl = dux_dxi*xizl + dux_dgamma*gammazl
378 
379  if (axisym .and. is_on_the_axis(ispec) .and. i == 1) then ! d_uz/dr=0 on the axis
380  duz_dxl = 0.d0
381  endif
382 
383  ! implement anisotropy in 2D
384  sigma_xx = c11*dux_dxl + c13*duz_dzl + c15*(duz_dxl + dux_dzl)
385  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
386  if (c12 < 1.e-7 .or. c23 < 1.e-7) stop 'cannot compute pressure for an anisotropic material if c12 or c23 are zero'
387  sigma_yy = c12*dux_dxl + c23*duz_dzl + c25*(duz_dxl + dux_dzl)
388  sigma_zz = c13*dux_dxl + c33*duz_dzl + c35*(duz_dxl + dux_dzl)
389 
390  endif
391 
392  ! store pressure
393  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
394  pressure_element(i,j) = - (sigma_xx + sigma_yy + sigma_zz) / 3.d0
395 
396  enddo
397  enddo
398 
399  else if (ispec_is_poroelastic(ispec)) then
400  ! poro-elastic element
401 
402  lambdal_unrelaxed_elastic = poroelastcoef(1,1,kmato(ispec))
403  mul_unrelaxed_elastic = poroelastcoef(2,1,kmato(ispec))
404 
405  ! get poroelastic parameters of current spectral element
406  call get_poroelastic_material(ispec,phi,tort,mu_s,kappa_s,rho_s,kappa_f,rho_f,eta_f,mu_fr,kappa_fr,rho_bar)
407 
408  ! Biot coefficients for the input phi
409  call get_poroelastic_biot_coeff(phi,kappa_s,kappa_f,kappa_fr,mu_fr,d_biot,h_biot,c_biot,m_biot)
410 
411  ! where T = G:grad u_s + C div w I
412  ! and T_f = C div u_s I + M div w I
413  ! we are expressing lambdaplus2mu, lambda, and mu for G, C, and M
414  mu_g = mu_fr
415  lambdal_g = h_biot - two*mu_fr
416  lambdalplus2mul_g = lambdal_g + two*mu_g
417 
418  do j = 1,ngllz
419  do i = 1,ngllx
420 
421  ! derivative along x and along z
422  dux_dxi = zero
423  duz_dxi = zero
424 
425  dux_dgamma = zero
426  duz_dgamma = zero
427 
428  dwx_dxi = zero
429  dwz_dxi = zero
430 
431  dwx_dgamma = zero
432  dwz_dgamma = zero
433 
434  ! first double loop over GLL points to compute and store gradients
435  ! we can merge the two loops because NGLLX == NGLLZ
436  do k = 1,ngllx
437  dux_dxi = dux_dxi + displs_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
438  duz_dxi = duz_dxi + displs_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
439  dux_dgamma = dux_dgamma + displs_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
440  duz_dgamma = duz_dgamma + displs_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
441 
442  dwx_dxi = dwx_dxi + displw_poroelastic(1,ibool(k,j,ispec))*hprime_xx(i,k)
443  dwz_dxi = dwz_dxi + displw_poroelastic(2,ibool(k,j,ispec))*hprime_xx(i,k)
444  dwx_dgamma = dwx_dgamma + displw_poroelastic(1,ibool(i,k,ispec))*hprime_zz(j,k)
445  dwz_dgamma = dwz_dgamma + displw_poroelastic(2,ibool(i,k,ispec))*hprime_zz(j,k)
446  enddo
447 
448  xixl = xix(i,j,ispec)
449  xizl = xiz(i,j,ispec)
450  gammaxl = gammax(i,j,ispec)
451  gammazl = gammaz(i,j,ispec)
452 
453  ! derivatives of displacement
454  dux_dxl = dux_dxi*xixl + dux_dgamma*gammaxl
455  duz_dzl = duz_dxi*xizl + duz_dgamma*gammazl
456 
457  dwx_dxl = dwx_dxi*xixl + dwx_dgamma*gammaxl
458  dwz_dzl = dwz_dxi*xizl + dwz_dgamma*gammazl
459 
460 ! compute diagonal components of the stress tensor (include attenuation if needed)
461 
462  if (attenuation_viscoelastic) then
463 
464 ! attenuation is implemented following the memory variable formulation of
465 ! J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
466 ! vol. 58(1), p. 110-120 (1993). More details can be found in
467 ! J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation in a linear
468 ! viscoelastic medium, Geophysical Journal International, vol. 95, p. 597-611 (1988).
469 
470 ! When implementing viscoelasticity according to the Carcione 1993 paper, attenuation is
471 ! non-causal rather than causal i.e. wave speed up instead of slowing down
472 ! when attenuation is turned on. We fixed that issue (which is not incorrect but non traditional)
473 ! by taking the unrelaxed state (infinite frequency) as a reference instead of the relaxed state (zero frequency)
474 ! and also using equations in Carcione's 2007 book.
475 ! See file doc/old_problem_attenuation_reference_Specfem2D_fixed_by_Xie_Zhinan.pdf
476 ! and doc/how_we_modified_Carcione_1993_to_make_it_causal_and_include_the_missing_1_over_L_factor.pdf
477 
478 ! See also J. M. Carcione, Seismic modeling in viscoelastic media, Geophysics,
479 ! vol. 58(1), p. 110-120 (1993) for two memory-variable mechanisms (page 112).
480 
481 ! and J. M. Carcione, D. Kosloff and R. Kosloff, Wave propagation simulation
482 ! in a linear viscoelastic medium, Geophysical Journal International,
483 ! vol. 95, p. 597-611 (1988) for two memory-variable mechanisms (page 604).
484 
485  ! compute the stress using the unrelaxed Lame parameters (Carcione 2007 page 125)
486  sigma_xx = (lambdal_unrelaxed_elastic + 2.0 * mul_unrelaxed_elastic)*dux_dxl + lambdal_unrelaxed_elastic*duz_dzl
487  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
488 ! sigma_yy = ... ! it is not zero because of the plane strain formulation, thus it should be computed here
489  stop 'pressure calculation not implemented for poroelastic media yet, you should compute sigma_yy here'
490  sigma_zz = (lambdal_unrelaxed_elastic + 2.0 * mul_unrelaxed_elastic)*duz_dzl + lambdal_unrelaxed_elastic*dux_dxl
491 
492  ! add the memory variables using the relaxed parameters (Carcione 2007 page 125)
493  ! beware: there is a bug in Carcione's equation (2c) of his 1993 paper for sigma_zz, we fixed it in the code below
494  e1_sum = 0._custom_real
495  e11_sum = 0._custom_real
496 
497  do i_sls = 1,n_sls
498  e1_sum = e1_sum + e1(i,j,ispec,i_sls)
499  e11_sum = e11_sum + e11(i,j,ispec,i_sls)
500  enddo
501 
502  sigma_xx = sigma_xx + (lambdal_unrelaxed_elastic + mul_unrelaxed_elastic) * e1_sum &
503  + two * mul_unrelaxed_elastic * e11_sum
504  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
505 ! sigma_yy = ... ! it is not zero because of the plane strain formulation, thus it should be computed here
506  stop 'pressure calculation not implemented for poroelastic media yet, you should compute sigma_yy here'
507  sigma_zz = sigma_zz + (lambdal_unrelaxed_elastic + mul_unrelaxed_elastic) * e1_sum &
508  - two * mul_unrelaxed_elastic * e11_sum
509 
510  else
511 
512  ! no attenuation
513  sigma_xx = lambdalplus2mul_g*dux_dxl + lambdal_g*duz_dzl + c_biot*(dwx_dxl + dwz_dzl)
514  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
515 ! sigma_yy = ... ! it is not zero because of the plane strain formulation, thus it should be computed here
516  stop 'pressure calculation not implemented for poroelastic media yet, you should compute sigma_yy here'
517  sigma_zz = lambdalplus2mul_g*duz_dzl + lambdal_g*dux_dxl + c_biot*(dwx_dxl + dwz_dzl)
518 
519 ! sigmap = C_biot*(dux_dxl + duz_dzl) + M_biot*(dwx_dxl + dwz_dzl)
520 
521  endif
522 
523  ! store pressure
524  ! sigma_yy is not equal to zero in a 2D medium because of the plane strain formulation
525  pressure_element(i,j) = - (sigma_xx + sigma_yy + sigma_zz) / 3.d0
526 ! pressure_element2(i,j) = - sigmap
527  enddo
528  enddo
529 
530 ! pressure = - Chi_dot_dot if acoustic element
531  else if (ispec_is_acoustic(ispec)) then
532  ! acoustic element
533 
534  if (use_trick_for_better_pressure) then
535  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
536  ! use the second derivative of the source for the source time function instead of the source itself,
537  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
538  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
539  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,
540  ! thus in fluid elements potential_dot_dot_acoustic() is accurate at zeroth order while potential_acoustic()
541  ! is accurate at second order and thus contains significantly less numerical noise.
542  ! compute pressure on the fluid/porous medium edge
543  do j = 1,ngllz
544  do i = 1,ngllx
545  iglob = ibool(i,j,ispec)
546  ! store pressure
547  pressure_element(i,j) = - potential_acoustic(iglob)
548  enddo
549  enddo
550  else
551  do j = 1,ngllz
552  do i = 1,ngllx
553  iglob = ibool(i,j,ispec)
554  ! store pressure
555  pressure_element(i,j) = - potential_dot_dot_acoustic(iglob)
556  enddo
557  enddo
558  endif
559 
560  else if (ispec_is_gravitoacoustic(ispec)) then
561  ! gravito-acoustic element
562 
563  if (use_trick_for_better_pressure) then
564  ! use a trick to increase accuracy of pressure seismograms in fluid (acoustic) elements:
565  ! use the second derivative of the source for the source time function instead of the source itself,
566  ! and then record -potential_acoustic() as pressure seismograms instead of -potential_dot_dot_acoustic();
567  ! this is mathematically equivalent, but numerically significantly more accurate because in the explicit
568  ! Newmark time scheme acceleration is accurate at zeroth order while displacement is accurate at second order,
569  ! thus in fluid elements potential_dot_dot_acoustic() is accurate at zeroth order while potential_acoustic()
570  ! is accurate at second order and thus contains significantly less numerical noise.
571  ! compute pressure on the fluid/porous medium edge
572  do j = 1,ngllz
573  do i = 1,ngllx
574  iglob = ibool(i,j,ispec)
575  ! store pressure
576  pressure_element(i,j) = - potential_gravitoacoustic(iglob)
577  enddo
578  enddo
579  else
580  do j = 1,ngllz
581  do i = 1,ngllx
582  iglob = ibool(i,j,ispec)
583  ! store pressure
584  pressure_element(i,j) = - potential_dot_dot_gravitoacoustic(iglob)
585  enddo
586  enddo
587  endif
588 
589  endif ! end of test if acoustic or elastic or gravito element
590 
591  end subroutine compute_pressure_one_element
