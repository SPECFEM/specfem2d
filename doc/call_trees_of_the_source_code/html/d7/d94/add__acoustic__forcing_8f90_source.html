65  xgamma = - xiz(i,j,ispec) * jacobian(i,j,ispec)
66  zgamma = + xix(i,j,ispec) * jacobian(i,j,ispec)
67  jacobian1d = sqrt(xgamma**2 + zgamma**2)
68  nx = - zgamma / jacobian1d
69  nz = + xgamma / jacobian1d
70  weight = jacobian1d * wzgll(j)
71 
72  ! define displacement components which will force the boundary
73  if (pml_boundary_conditions) then
74  if (ispec_is_pml(ispec)) then
75  displ_x = 0.0d0
76  displ_z = 0.0d0
77  else
78  call acoustic_forcing_boundary(iglob,displ_x,displ_z)
79  endif
80  else
81  call acoustic_forcing_boundary(iglob,displ_x,displ_z)
82  endif
83  ! compute dot product
84  displ_n = displ_x*nx + displ_z*nz
85  potential_dot_dot_acoustic(iglob) = potential_dot_dot_acoustic(iglob) + weight*displ_n
86  enddo
87  endif ! end of left acoustic forcing boundary
88 
89  !--- right acoustic forcing boundary
90  if (codeacforcing(iedge2,inum)) then

108  endif

118  endif ! end of right acoustic forcing boundary

122  j = 1

124  iglob = ibool(i,j,ispec)
125  xxi = + gammaz(i,j,ispec) * jacobian(i,j,ispec)
126  zxi = - gammax(i,j,ispec) * jacobian(i,j,ispec)

270  gravityl = gravityext(i,j,ispec)

278  (tempx1l*xixl + tempx2l*gammaxl)*nx ) / (0._custom_real - gravityl*nz)

287  enddo

307  else
308  call acoustic_forcing_boundary(iglob,displ_x,displ_z)
309  endif
