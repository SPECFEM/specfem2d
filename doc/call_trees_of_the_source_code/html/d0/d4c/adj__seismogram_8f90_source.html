34 program adj_seismogram
35 
36 ! This program cuts a certain portion of the seismograms and convert it
37 ! into the adjoint source for generating banana-dougnut kernels
38 !
39 ! The calculated adjoint source corresponds to a (cross-correlation) traveltime measurement
40 ! within a given time window
41 !
42 ! compilation:
43 ! rm -rf xadj_seismogram ; gfortran adj_seismogram.f90 -o xadj_seismogram
44 !
45 ! usage example:
46 ! ./xadj_seismogram 27.0 32. AA.S0001 1
47 
48  implicit none
49 
50  ! number of time steps
51  integer :: NSTEP
52 
53  ! time step size
54  double precision :: deltat
55 
56  integer :: itime,icomp,istart,iend,nlen,irec,i
57  integer :: NDIM,NDIMr,adj_comp
58  integer :: ier
59 
60  double precision :: t0,time,time0,dummy,Nnorm
61 
62  double precision, dimension(:),allocatable :: time_window
63  double precision, dimension(:,:),allocatable :: seism
64  double precision, dimension(:),allocatable :: seism_veloc,seism_accel,ft_bar,seism_win
65 
66  character(len=3) :: compr(2),comp(3)
67  character(len=150) :: filename
68 
69  ! single station setup
70  integer, parameter :: nrec = 1
71  double precision,dimension(nrec) :: tstart(nrec),tend(nrec)
72  character(len=150), dimension(nrec) :: station_name
73 
74  ! tolerance
75  double precision, parameter :: EPS = 1.d-40
76 
77  ! input
78  character(len=256) :: arg
79 
80 !--------------------------------------------------------------------------------
81 ! USER PARAMETERS
82 !--------------------------------------------------------------------------------
83 
84  ! reads in file arguments
85  do i = 1,4
86  call get_command_argument(i,arg)
87  if (trim(arg) == '' .or. command_argument_count() /= 4) then
88  print *,'Usage: '
89  print *,' xadj_seismogram t1 t2 station-name adj_comp[1-3]'
90  print *,'with'
91  print *,' t1: window start time'
92  print *,' t2: window end time'
93  print *,' station-name : adjoint station (e.g. AA.S00001)'
94  print *,' adj_comp: 1 = adjoint source given by X component'
95  print *,' adj_comp: 2 = adjoint source given by Y component (SH waves)'
96  print *,' adj_comp: 3 = adjoint source given by Z component'
97  stop 1
98  endif
99  select case (i)
100  case (1)
101  read(arg,*,iostat=ier) tstart(1)
102  if (ier /= 0) stop 'Error reading tstart'
103  case (2)
104  read(arg,*,iostat=ier) tend(1)
105  if (ier /= 0) stop 'Error reading tend'
106  case (3)
107  station_name(1) = trim(arg)
108  case (4)
109  read(arg,*,iostat=ier) adj_comp
110  if (ier /= 0) stop 'Error reading adjoint component'
111  end select
112  enddo
113 
114  ! checks input arguments
115  if (adj_comp /= 1 .and. adj_comp /= 2 .and. adj_comp /= 3) then
116  print *,'Invalid adj_comp number, must be 1 (X-comp), 2 (Y-comp) or 3 (Z-comp)'
117  stop 1
118  endif
119 
120  !---------------------------------
121  ! trace component setup
122  !---------------------------------
123  ! number of components
124  !NDIMr = 2 ! P-SV
125  !NDIMr = 1 ! SH (membrane)
126 
127  !---------------------------------
128  ! single station setup
129  !---------------------------------
130  ! list of stations
131  !station_name(1) = 'AA.'//'S0001'
132 
133  ! KEY: 'absolute' time interval for window
134  !tstart(1) = 27.0 + 8.0
135  !tend(1) = 32.0 + 8.0
136 
137  ! chose the component for the adjoint source (adj_comp = 1:X, 2:Y, 3:Z)
138  !adj_comp = 1
139  !---------------------------------
140 
141  ! all components
142  ndim = 3
143  comp = (/"BXX","BXY","BXZ"/)
144 
145  ! select number of dimensions of input trace
146  if (adj_comp == 2) then
147  ! Y-component only for SH-waves
148  ndimr = 1
149  else
150  ! X- or Z-component for P-SV waves
151  ndimr = 2
152  endif
153 
154  ! select wave type components
155  if (ndimr == 1) then
156  ! SH (membrane)
157  compr = (/"BXY","tmp"/)
158  else
159  ! P-SV
160  compr = (/"BXX","BXZ"/)
161  endif
162 
163  ! user output
164  print *,'adjoint source - seismogram:'
165  print *
166  print *,'setup:'
167  print *,' number of adjoint stations (nrec) = ',nrec
168  do irec = 1,nrec
169  print *,' station name ',irec,': ',station_name(irec)
170  enddo
171  print *
172  print *,' seismogram components = ',ndimr
173  if (ndimr == 1) then
174  print *,' seismogram label = ',compr(1)
175  else
176  print *,' seismogram labels = ',compr(1),' / ',compr(2)
177  endif
178  print *
179  print *,' time window start/end = ',tstart(1),tend(1)
180  print *,' adjoint source trace component (1 == X / 2 == Y / 3 == Z) = ',adj_comp
181  print *
182 
183  ! basic check
184  do irec = 1,nrec
185  if (tstart(irec) >= tend(irec)) then
186  print *,"Invalid start/end time chosen for station ",irec,": tstart = ",tstart(irec), &
187  " should be less then tend = ",tend(irec)
188  stop 1
189  endif
190  enddo
191 
192  ! initializes
193  nstep = 0
194  t0 = 0.d0
195 
196  ! gets trace length and start time t0
197  do irec = 1,nrec
198 
199  do icomp = 1,ndimr
200 
201  filename = 'OUTPUT_FILES/'//trim(station_name(irec))//'.'// compr(icomp) // '.semd'
202  open(unit = 10, file = trim(filename),status='old',iostat=ier)
203  if (ier /= 0) then
204  print *,'Error opening file: ',trim(filename)
205  !stop 'Error opening trace file' ! note: stop '**comment**' finishes with an error code 0 on some systems
206  stop 10
207  endif
208 
209  ! counts lines
210  nlen = 0
211  do while (ier == 0)
212  read(10,*,iostat=ier) time , dummy
213  if (ier == 0) nlen = nlen + 1
214  enddo
215  rewind(10)
216 
217  ! sets length NSTEP from first file
218  if (irec == 1 .and. icomp == 1) then
219  nstep = nlen
220  else
221  ! checks with settings
222  if (nlen /= nstep) then
223  print *,'Error: number of lines ',nlen,' in file ',trim(filename),' do not match setting NSTEP = ',nstep
224  stop 20
225  endif
226  endif
227 
228  ! check
229  if (nstep <= 1) then
230  print *,'Error: invalid number of lines (time steps) encountered: NSTEP = ',nstep
231  stop 30
232  endif
233 
234  ! sets time step size
235  do itime = 1,nstep
236  read(10,*) time , dummy
237 
238  ! stores start time and time step size
239  if (irec == 1 .and. icomp == 1) then
240  ! sets start time
241  if (itime == 1) t0 = time
242  ! sets time step
243  if (itime == nstep) deltat = (time - t0)/dble(nstep-1)
244  else
245  ! checks start time
246  if (itime == 1) then
247  time0 = time
248  if (abs(t0 - time0) > 1.e-5) then
249  print *,'Error: start time',time0,' in file ',trim(filename),' does not match t0 = ',t0
250  stop 40
251  endif
252  endif
253 
254  ! checks time step size
255  if (itime == nstep) then
256  if (abs( (time - time0)/dble(nstep-1) - deltat) > 1.e-5) then
257  print *,'Error: time step size ',(time-time0)/dble(nstep-1),' in file ',trim(filename), &
258  ' does not match deltat = ',deltat
259  stop 50
260  endif
261  endif
262  endif
263  enddo
264  close(10)
265 
266  enddo
267  enddo
268 
269  ! user output
270  print *,'reading input traces:'
271  print *,' number of time steps (NSTEP) = ',nstep
272  print *
273  print *,' start time (t0) = ',t0
274  print *,' end time = ',(nstep-1)*deltat + t0
275  print *,' time step size (deltat) = ',deltat
276  print *
277 
278  ! allocates trace arrays
279  allocate( time_window(nstep), &
280  seism(nstep,3), &
281  seism_win(nstep), &
282  seism_veloc(nstep), &
283  seism_accel(nstep), &
284  ft_bar(nstep),stat=ier)
285  if (ier /= 0) stop 2
286 
287  ! creates adjoint sources
288  do irec = 1,nrec
289 
290  ! reads in all trace components
291  do icomp = 1,ndimr
292 
293  filename = 'OUTPUT_FILES/'//trim(station_name(irec))//'.'// compr(icomp) // '.semd'
294  open(unit = 10, file = trim(filename),status='old',iostat=ier)
295  if (ier /= 0) then
296  print *,'Error opening file: ',trim(filename)
297  stop 'Error opening trace file'
298  endif
299 
300  ! reads in trace
301  do itime = 1,nstep
302  read(10,*) time , seism(itime,icomp)
303  enddo
304  close(10)
305  enddo
306 
307  if (ndimr == 2) then
308  seism(:,3) = seism(:,2)
309  seism(:,2) = 0.d0
310  else
311  seism(:,2) = seism(:,1)
312  seism(:,1) = 0.d0
313  seism(:,3) = 0.d0
314  endif
315 
316  ! start/end index
317  istart = floor((tstart(irec) - t0)/deltat) + 1
318  iend = ceiling((tend(irec) - t0)/deltat) + 1
319 
320  ! user output
321  if (istart < 1 .or. istart > nstep -1) &
322  print *,'*** warning: time window start is out of trace length! start time will be moved ***'
323  if (iend < 1 .or. iend > nstep) &
324  print *,'*** warning: time window end is out of trace length! end time will be moved ***'
325 
326  ! limits
327  if (istart < 1) istart = 1
328  if (istart > nstep) istart = nstep - 1
329  if (iend < istart) iend = istart + 1
330  if (iend > nstep) iend = nstep
331 
332  ! user output
333  print *,'time window:'
334  print *,' index : istart =',istart, 'iend =', iend
335  print *,' time window: tstart =',(istart-1)*deltat + t0, 'tend =',(iend-1)*deltat + t0
336  print *
337 
338  if (istart >= iend) then
339  print *,"Error start/end index: ",istart,iend
340  stop 11
341  endif
342 
343  ! window length
344  nlen = iend - istart + 1
345 
346  ! outputs all 3-component X/Y/Z (needed for kernel simulations)
347  do icomp = 1, ndim
348 
349  print *,'component: ',comp(icomp)
350 
351  filename = 'SEM/'//trim(station_name(irec))//'.'// comp(icomp) // '.adj'
352  open(unit = 11, file = trim(filename),status='unknown',iostat=ier)
353  if (ier /= 0) then
354  print *,'Error opening file: ',trim(filename)
355  stop 60
356  endif
357 
358  time_window(:) = 0.d0
359  seism_win(:) = seism(:,icomp)
360  seism_veloc(:) = 0.d0
361  seism_accel(:) = 0.d0
362 
363  do itime = istart,iend
364  ! cosine window
365  !time_window(itime) = 1.d0 - cos(pi*(itime-1)/NSTEP+1)**10
366 
367  ! Welch window
368  time_window(itime) = 1.d0 - (2* (dble(itime) - istart)/(iend-istart) -1.d0)**2
369  enddo
370 
371  ! first time derivative (by finite-differences)
372  do itime = 2,nstep-1
373  seism_veloc(itime) = (seism_win(itime+1) - seism_win(itime-1))/(2*deltat)
374  enddo
375  seism_veloc(1) = (seism_win(2) - seism_win(1))/deltat
376  seism_veloc(nstep) = (seism_win(nstep) - seism_win(nstep-1))/deltat
377 
378  ! second time derivative
379  do itime = 2,nstep-1
380  seism_accel(itime) = (seism_veloc(itime+1) - seism_veloc(itime-1))/(2*deltat)
381  enddo
382  seism_accel(1) = (seism_veloc(2) - seism_veloc(1))/deltat
383  seism_accel(nstep) = (seism_veloc(nstep) - seism_veloc(nstep-1))/deltat
384 
385  ! cross-correlation traveltime adjoint source
386  nnorm = deltat * sum(time_window(:) * seism_win(:) * seism_accel(:))
387 
388  !Nnorm = deltat * sum(time_window(:) * seism_veloc(:) * seism_veloc(:))
389 
390  if (abs(nnorm) > eps) then
391  !ft_bar(:) = -seism_veloc(:) * time_window(:) / Nnorm
392  ft_bar(:) = seism_veloc(:) * time_window(:) / nnorm
393  print *,'Norm =', nnorm
394  else
395  print *,'Norm < EPS for file, zeroeing out trace'
396  print *,'Norm =', nnorm
397  ft_bar(:) = 0.d0
398  endif
399  print *
400 
401  do itime = 1,nstep
402  if (icomp == adj_comp) then
403  write(11,*) (itime-1)*deltat - t0, ft_bar(itime)
404  else
405  write(11,*) (itime-1)*deltat - t0, 0.d0
406  endif
407  enddo
408  close(11)
409 
410  enddo
411 
412  enddo
413 
414  ! user output
415  print *,'*************************'
416  print *,'The input files (AA.S****.BXX/BXY/BXZ.adj) needed to run the adjoint simulation are in folder SEM/'
417  print *,'*************************'
418 
419 end program adj_seismogram
