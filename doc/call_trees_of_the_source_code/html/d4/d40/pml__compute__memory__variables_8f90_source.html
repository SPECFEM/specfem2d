34 ! calculates C-PML acoustic/elastic memory variables and computes stress sigma
35 
36 ! second-order accurate convolution term calculation from equation (21) of
37 ! Shumin Wang, Robert Lee, and Fernando L. Teixeira,
38 ! Anisotropic-medium PML for vector FETD with modified basis functions,
39 ! IEEE Transactions on Antennas and Propagation, vol. 54, no. 1, (2006)
40 
41 
42  subroutine pml_compute_memory_variables_acoustic(ispec,nglob,potential_acoustic_old,dux_dxl,dux_dzl)
43 
44 ! for acoustic elements
45 
46  use constants, only: custom_real,ngllx,ngllz,cpml_x_only,cpml_z_only,alpha_lddrk,beta_lddrk,c_lddrk
47 
48  use specfem_par, only: time_stepping_scheme,i_stage,it,deltat, &
49  ibool,xix,xiz,gammax,gammaz, &
50  hprime_xx,hprime_zz, &
51  rmemory_acoustic_dux_dx,rmemory_acoustic_dux_dz, &
52  rmemory_acoustic_dux_dx_lddrk,rmemory_acoustic_dux_dz_lddrk, &
53  axisym,is_on_the_axis,hprimebar_xx
54 
55  ! PML arrays
56  use specfem_par, only: nspec_pml,ispec_is_pml,spec_to_pml,region_cpml, &
57  k_x_store,k_z_store,d_x_store,d_z_store,alpha_x_store,alpha_z_store
58 
59  implicit none
60 
61  integer,intent(in) :: ispec
62 
63  integer,intent(in) :: nglob
64  real(kind=CUSTOM_REAL), dimension(nglob),intent(in) :: potential_acoustic_old
65 
66  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(inout) :: dux_dxl,dux_dzl
67 
68  ! local parameters
69  integer :: i,j,k
70  real(kind=CUSTOM_REAL) :: dux_dxi,dux_dgamma
71  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl
72 
73  ! PML
74  integer :: ispec_PML
75  integer :: CPML_region_local
76  integer :: singularity_type_zx,singularity_type_xz
77  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: PML_dux_dxl,PML_dux_dzl,PML_dux_dxl_old,PML_dux_dzl_old
78 
79  double precision :: kappa_x,kappa_z,d_x,d_z,alpha_x,alpha_z,beta_x,beta_z,time_n,time_nsub1
80  double precision :: A5,A6,A7
81  double precision :: bb_zx_1,bb_zx_2
82  double precision :: coef0_zx_1,coef1_zx_1,coef2_zx_1,coef0_zx_2,coef1_zx_2,coef2_zx_2
83  double precision :: A8,A9,A10
84  double precision :: bb_xz_1,bb_xz_2
85  double precision :: coef0_xz_1,coef1_xz_1,coef2_xz_1,coef0_xz_2,coef1_xz_2,coef2_xz_2
86 
87  ! checks if anything to do in this slice
88  if (nspec_pml == 0) return
89  if (.not. ispec_is_pml(ispec)) return
90 
91  do j = 1,ngllz
92  do i = 1,ngllx
93  ! stores initial derivatives
94  pml_dux_dxl(i,j) = dux_dxl(i,j)
95  pml_dux_dzl(i,j) = dux_dzl(i,j)
96 
97  select case (time_stepping_scheme)
98  case (1)
99  ! Newmark
100  dux_dxi = 0._custom_real
101  dux_dgamma = 0._custom_real
102 
103  do k = 1,ngllx
104  dux_dxi = dux_dxi + potential_acoustic_old(ibool(k,j,ispec)) * hprime_xx(i,k)
105  dux_dgamma = dux_dgamma + potential_acoustic_old(ibool(i,k,ispec)) * hprime_zz(j,k)
106  enddo
107 
108  ! AXISYM overwrites dux_dxi
109  if (axisym) then
110  if (is_on_the_axis(ispec)) then
111  dux_dxi = 0._custom_real
112  do k = 1,ngllx
113  dux_dxi = dux_dxi + potential_acoustic_old(ibool(k,j,ispec)) * hprimebar_xx(i,k)
114  enddo
115  endif
116  endif
117 
118  ! derivatives of potential
119  xixl = xix(i,j,ispec)
120  xizl = xiz(i,j,ispec)
121  gammaxl = gammax(i,j,ispec)
122  gammazl = gammaz(i,j,ispec)
123 
124  pml_dux_dxl_old(i,j) = dux_dxi * xixl + dux_dgamma * gammaxl
125  pml_dux_dzl_old(i,j) = dux_dxi * xizl + dux_dgamma * gammazl
126 
127 ! if (AXISYM) then ! TODO LDDRK
128 ! if (is_on_the_axis(ispec) .and. i == 1) then
129 ! ! dchi/dr=rho * u_r=0 on the axis
130 ! PML_dux_dxl_old(i,j) = 0._CUSTOM_REAL
131 ! endif
132 ! endif
133  end select
134  enddo
135  enddo
136 
137  ! timing
138  select case (time_stepping_scheme)
139  case (1)
140  ! Newmark
141  time_n = (it-1) * deltat
142  time_nsub1 = (it-2) * deltat
143  case (2)
144  ! LDDRK
145  time_n = (it-1) * deltat + c_lddrk(i_stage) * deltat
146  case default
147  stop 'Sorry, time stepping scheme not implemented yet in PML memory variable updates'
148  end select
149 
150  ! local PML element
151  ispec_pml = spec_to_pml(ispec)
152  cpml_region_local = region_cpml(ispec)
153 
154  do j = 1,ngllz
155  do i = 1,ngllx
156  kappa_x = k_x_store(i,j,ispec_pml)
157  kappa_z = k_z_store(i,j,ispec_pml)
158 
159  d_x = d_x_store(i,j,ispec_pml)
160  d_z = d_z_store(i,j,ispec_pml)
161 
162  alpha_x = alpha_x_store(i,j,ispec_pml)
163  alpha_z = alpha_z_store(i,j,ispec_pml)
164 
165  beta_x = alpha_x + d_x / kappa_x
166  beta_z = alpha_z + d_z / kappa_z
167 
168  ! gets PML coefficients
169  ! the subroutine of lik_parameter_computation is located at the end of compute_forces_viscoelastic.F90
170  call lik_parameter_computation(time_n,deltat,kappa_z,beta_z,alpha_z,kappa_x,beta_x,alpha_x, &
171  cpml_region_local,31,a5,a6,a7,singularity_type_zx,bb_zx_1,bb_zx_2, &
172  coef0_zx_1,coef1_zx_1,coef2_zx_1,coef0_zx_2,coef1_zx_2,coef2_zx_2)
173 
174  call lik_parameter_computation(time_n,deltat,kappa_x,beta_x,alpha_x,kappa_z,beta_z,alpha_z, &
175  cpml_region_local,13,a8,a9,a10,singularity_type_xz,bb_xz_1,bb_xz_2, &
176  coef0_xz_1,coef1_xz_1,coef2_xz_1,coef0_xz_2,coef1_xz_2,coef2_xz_2)
177 
178  select case (time_stepping_scheme)
179  case (1)
180  ! Newmark
181  rmemory_acoustic_dux_dx(i,j,ispec_pml,1) = coef0_zx_1 * rmemory_acoustic_dux_dx(i,j,ispec_pml,1) + &
182  coef1_zx_1 * pml_dux_dxl(i,j) + coef2_zx_1 * pml_dux_dxl_old(i,j)
183  if (singularity_type_zx == 0) then
184  rmemory_acoustic_dux_dx(i,j,ispec_pml,2) = coef0_zx_2 * rmemory_acoustic_dux_dx(i,j,ispec_pml,2) + &
185  coef1_zx_2 * pml_dux_dxl(i,j) + coef2_zx_2 * pml_dux_dxl_old(i,j)
186  else
187  rmemory_acoustic_dux_dx(i,j,ispec_pml,2) = coef0_zx_2 * rmemory_acoustic_dux_dx(i,j,ispec_pml,2) + &
188  coef1_zx_2 * time_n * pml_dux_dxl(i,j) + &
189  coef2_zx_2 * time_nsub1 * pml_dux_dxl_old(i,j)
190  endif
191 
192  rmemory_acoustic_dux_dz(i,j,ispec_pml,1) = coef0_xz_1 * rmemory_acoustic_dux_dz(i,j,ispec_pml,1) + &
193  coef1_xz_1 * pml_dux_dzl(i,j) + coef2_xz_1 * pml_dux_dzl_old(i,j)
194  if (singularity_type_xz == 0) then
195  rmemory_acoustic_dux_dz(i,j,ispec_pml,2) = coef0_xz_2 * rmemory_acoustic_dux_dz(i,j,ispec_pml,2) + &
196  coef1_xz_2 * pml_dux_dzl(i,j) + coef2_xz_2 * pml_dux_dzl_old(i,j)
197  else
198  rmemory_acoustic_dux_dz(i,j,ispec_pml,2) = coef0_xz_2 * rmemory_acoustic_dux_dz(i,j,ispec_pml,2) + &
199  coef1_xz_2 * time_n * pml_dux_dzl(i,j) + &
200  coef2_xz_2 * time_nsub1 * pml_dux_dzl_old(i,j)
201  endif
202 
203  case (2)
204  ! LDDRK
205  rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,1) = &
206  alpha_lddrk(i_stage) * rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,1) + &
207  deltat * (-bb_zx_1 * rmemory_acoustic_dux_dx(i,j,ispec_pml,1) + pml_dux_dxl(i,j))
208  rmemory_acoustic_dux_dx(i,j,ispec_pml,1) = rmemory_acoustic_dux_dx(i,j,ispec_pml,1) + &
209  beta_lddrk(i_stage) * rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,1)
210 
211  if (singularity_type_zx == 0) then
212  rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,2) = &
213  alpha_lddrk(i_stage) * rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,2) + &
214  deltat * (-bb_zx_2 * rmemory_acoustic_dux_dx(i,j,ispec_pml,2) + pml_dux_dxl(i,j))
215  rmemory_acoustic_dux_dx(i,j,ispec_pml,2) = rmemory_acoustic_dux_dx(i,j,ispec_pml,2) + &
216  beta_lddrk(i_stage) * rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,2)
217  else
218  rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,2) = &
219  alpha_lddrk(i_stage) * rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,2) + &
220  deltat * (-bb_zx_2 * rmemory_acoustic_dux_dx(i,j,ispec_pml,2) + pml_dux_dxl(i,j) * time_n)
221  rmemory_acoustic_dux_dx(i,j,ispec_pml,2) = rmemory_acoustic_dux_dx(i,j,ispec_pml,2) + &
222  beta_lddrk(i_stage) * rmemory_acoustic_dux_dx_lddrk(i,j,ispec_pml,2)
223  endif
224 
225  rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,1) = &
226  alpha_lddrk(i_stage) * rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,1) + &
227  deltat * (-bb_xz_1 * rmemory_acoustic_dux_dz(i,j,ispec_pml,1) + pml_dux_dzl(i,j))
228  rmemory_acoustic_dux_dz(i,j,ispec_pml,1) = rmemory_acoustic_dux_dz(i,j,ispec_pml,1) + &
229  beta_lddrk(i_stage) * rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,1)
230 
231  if (singularity_type_xz == 0) then
232  rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,2) = &
233  alpha_lddrk(i_stage) * rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,2) + &
234  deltat * (-bb_xz_2 * rmemory_acoustic_dux_dz(i,j,ispec_pml,2) + pml_dux_dzl(i,j))
235  rmemory_acoustic_dux_dz(i,j,ispec_pml,2) = rmemory_acoustic_dux_dz(i,j,ispec_pml,2) + &
236  beta_lddrk(i_stage) * rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,2)
237  else
238  rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,2) = &
239  alpha_lddrk(i_stage) * rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,2) + &
240  deltat * (-bb_xz_2 * rmemory_acoustic_dux_dz(i,j,ispec_pml,2) + pml_dux_dzl(i,j) * time_n)
241  rmemory_acoustic_dux_dz(i,j,ispec_pml,2) = rmemory_acoustic_dux_dz(i,j,ispec_pml,2) + &
242  beta_lddrk(i_stage) * rmemory_acoustic_dux_dz_lddrk(i,j,ispec_pml,2)
243  endif
244 
245  case default
246  stop 'Time stepping scheme not implemented yet for PML memory variables'
247  end select
248 
249  dux_dxl(i,j) = a5 * pml_dux_dxl(i,j) + a6 * rmemory_acoustic_dux_dx(i,j,ispec_pml,1) + &
250  a7 * rmemory_acoustic_dux_dx(i,j,ispec_pml,2)
251  dux_dzl(i,j) = a8 * pml_dux_dzl(i,j) + a9 * rmemory_acoustic_dux_dz(i,j,ispec_pml,1) + &
252  a10 * rmemory_acoustic_dux_dz(i,j,ispec_pml,2)
253  enddo
254  enddo
255 
256  end subroutine pml_compute_memory_variables_acoustic

262  subroutine pml_compute_memory_variables_elastic(ispec,nglob,displ_elastic_old, &
263  dux_dxl,dux_dzl,duz_dxl,duz_dzl, &
264  dux_dxl_prime,dux_dzl_prime,duz_dxl_prime,duz_dzl_prime, &
265  PML_dux_dxl,PML_dux_dzl,PML_duz_dxl,PML_duz_dzl, &
266  PML_dux_dxl_old,PML_dux_dzl_old,PML_duz_dxl_old,PML_duz_dzl_old)
267 
268 ! for elastic elements
269 
270  use constants, only: custom_real,ndim,ngllx,ngllz,nglj, &
271  cpml_x_only,cpml_z_only,alpha_lddrk,beta_lddrk,c_lddrk
272 
273  use specfem_par, only: time_stepping_scheme,i_stage,it,deltat, &
274  ibool,xix,xiz,gammax,gammaz, &
275  hprime_xx,hprime_zz, &
276  axisym,is_on_the_axis,hprimebar_xx
277 
278  use specfem_par, only: rmemory_dux_dx,rmemory_dux_dz,rmemory_duz_dx,rmemory_duz_dz, &
279  rmemory_dux_dx_prime,rmemory_dux_dz_prime,rmemory_duz_dx_prime,rmemory_duz_dz_prime, &
280  rmemory_dux_dx_lddrk,rmemory_dux_dz_lddrk,rmemory_duz_dx_lddrk,rmemory_duz_dz_lddrk
281 
282  ! PML arrays
283  use specfem_par, only: nspec_pml,ispec_is_pml,spec_to_pml,region_cpml, &
284  k_x_store,k_z_store,d_x_store,d_z_store,alpha_x_store,alpha_z_store, &
285  rotate_pml_activate
286 
287  implicit none
288 
289  integer,intent(in) :: ispec
290 
291  integer,intent(in) :: nglob
292  real(kind=CUSTOM_REAL), dimension(NDIM,nglob),intent(in) :: displ_elastic_old
293 
294  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(inout) :: dux_dxl,dux_dzl,duz_dxl,duz_dzl
295  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: dux_dxl_prime,dux_dzl_prime,duz_dxl_prime,duz_dzl_prime
296 
297  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: PML_dux_dxl,PML_dux_dzl,PML_duz_dxl,PML_duz_dzl
298  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ),intent(out) :: PML_dux_dxl_old,PML_dux_dzl_old,PML_duz_dxl_old,PML_duz_dzl_old
299 
300  ! local parameters
301  integer :: i,j,k
302  real(kind=CUSTOM_REAL) :: dux_dxi_old,dux_dgamma_old,duz_dxi_old,duz_dgamma_old
303  real(kind=CUSTOM_REAL) :: xixl,xizl,gammaxl,gammazl
304 
305  ! PML
306  integer :: ispec_PML
307  integer :: CPML_region_local
308  integer :: singularity_type_zx,singularity_type_xz
309  double precision :: time_n,time_nsub1
310  double precision :: kappa_x,kappa_z,d_x,d_z,alpha_x,alpha_z,beta_x,beta_z
311  double precision :: A5,A6,A7
312  double precision :: bb_zx_1,bb_zx_2,coef0_zx_1,coef1_zx_1,coef2_zx_1,coef0_zx_2,coef1_zx_2,coef2_zx_2
313  double precision :: A8,A9,A10
314  double precision :: bb_xz_1,bb_xz_2,coef0_xz_1,coef1_xz_1,coef2_xz_1,coef0_xz_2,coef1_xz_2,coef2_xz_2
315 
316  ! checks if anything to do in this slice
317  if (nspec_pml == 0) return
318  if (.not. ispec_is_pml(ispec)) return
319 
320  do j = 1,ngllz
321  do i = 1,ngllx
322  ! stores initial derivatives
323  pml_dux_dxl(i,j) = dux_dxl(i,j)
324  pml_dux_dzl(i,j) = dux_dzl(i,j)
325  pml_duz_dzl(i,j) = duz_dzl(i,j)
326  pml_duz_dxl(i,j) = duz_dxl(i,j)
327 
328  if (axisym) then ! TODO LDDRK
329  if (is_on_the_axis(ispec) .and. i == 1) then
330  ! d_uz/dr=0 on the axis
331  pml_duz_dxl(i,j) = 0._custom_real
332  endif
333  endif
334 
335  ! derivative along x and along z
336  !
337  ! first double loop over GLL points to compute and store gradients
338  ! we can merge the two loops because NGLLX == NGLLZ
339  dux_dxi_old = 0._custom_real
340  duz_dxi_old = 0._custom_real
341  dux_dgamma_old = 0._custom_real
342  duz_dgamma_old = 0._custom_real
343  do k = 1,ngllx
344  dux_dxi_old = dux_dxi_old + displ_elastic_old(1,ibool(k,j,ispec))*hprime_xx(i,k)
345  duz_dxi_old = duz_dxi_old + displ_elastic_old(2,ibool(k,j,ispec))*hprime_xx(i,k)
346  dux_dgamma_old = dux_dgamma_old + displ_elastic_old(1,ibool(i,k,ispec))*hprime_zz(j,k)
347  duz_dgamma_old = duz_dgamma_old + displ_elastic_old(2,ibool(i,k,ispec))*hprime_zz(j,k)
348  enddo
349 
350  ! AXISYM overwrite dux_dxi and duz_dxi
351  if (axisym) then
352  if (is_on_the_axis(ispec)) then
353  dux_dxi_old = 0._custom_real
354  duz_dxi_old = 0._custom_real
355  do k = 1,nglj
356  dux_dxi_old = dux_dxi_old + displ_elastic_old(1,ibool(k,j,ispec))*hprimebar_xx(i,k)
357  duz_dxi_old = duz_dxi_old + displ_elastic_old(2,ibool(k,j,ispec))*hprimebar_xx(i,k)
358  enddo
359  endif
360  endif
361 
362  ! derivatives of displacement
363  xixl = xix(i,j,ispec)
364  xizl = xiz(i,j,ispec)
365  gammaxl = gammax(i,j,ispec)
366  gammazl = gammaz(i,j,ispec)
367 
368  pml_dux_dxl_old(i,j) = dux_dxi_old*xixl + dux_dgamma_old*gammaxl !dux_dxl_old
369  pml_dux_dzl_old(i,j) = dux_dxi_old*xizl + dux_dgamma_old*gammazl !dux_dzl_old
370  pml_duz_dxl_old(i,j) = duz_dxi_old*xixl + duz_dgamma_old*gammaxl !duz_dxl_old
371  pml_duz_dzl_old(i,j) = duz_dxi_old*xizl + duz_dgamma_old*gammazl !duz_dzl_old
372 
373  if (axisym) then ! TODO LDDRK
374  if (is_on_the_axis(ispec) .and. i == 1) then
375  ! d_uz/dr=0 on the axis
376  pml_duz_dxl_old(i,j) = 0._custom_real
377  endif
378  endif
379  enddo
380  enddo
381 
382  !------------------------------------------------------------------------------
383  !---------------------------- LEFT & RIGHT ------------------------------------
384  !------------------------------------------------------------------------------
385  ! timing
386  select case (time_stepping_scheme)
387  case (1)
388  ! Newmark
389  time_n = (it-1) * deltat
390  time_nsub1 = (it-2) * deltat
391  case (2)
392  ! LDDRK
393  time_n = (it-1) * deltat + c_lddrk(i_stage) * deltat
394  case default
395  stop 'Sorry, time stepping scheme not implemented yet in PML memory variable updates'
396  end select
