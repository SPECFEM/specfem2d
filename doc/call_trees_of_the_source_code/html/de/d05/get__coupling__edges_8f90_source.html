34  subroutine get_coupling_edges()
35 
36 ! handles all domain coupling
37 
38  use constants, only: imain,ibottom,iright,itop,ileft,tinyval
39  use specfem_par
40 
41  implicit none
42 
43  ! local parameters
44  integer :: ispec_poroelastic,iedge_poroelastic
45  integer :: ispec_acoustic,ispec_elastic
46  integer :: iedge_acoustic,iedge_elastic
47  integer :: inum,ipoin1D,iglob2
48  integer :: i,j,iglob
49 
50  ! determine if coupled fluid-solid simulation
51  coupled_acoustic_elastic = any_acoustic .and. any_elastic
52  coupled_acoustic_poro = any_acoustic .and. any_poroelastic
53  coupled_elastic_poro = any_elastic .and. any_poroelastic
54 
55  ! fluid/solid (elastic) edge detection
56  ! the two elements (fluid and solid) forming an edge are already known (computed in meshfem2D),
57  ! the common nodes forming the edge are computed here
58  if (coupled_acoustic_elastic) then
59 
60  ! user output
61  if (myrank == 0) then
62  write(imain,*)
63  write(imain,*) 'Mixed acoustic/elastic simulation'
64  write(imain,*)
65  write(imain,*) ' Beginning of fluid/solid edge detection'
66  call flush_imain()
67  endif
68 
69  ! define the edges of a given element
70  i_begin(ibottom) = 1
71  j_begin(ibottom) = 1
72  i_end(ibottom) = ngllx
73  j_end(ibottom) = 1
74 
75  i_begin(iright) = ngllx
76  j_begin(iright) = 1
77  i_end(iright) = ngllx
78  j_end(iright) = ngllz
79 
80  i_begin(itop) = ngllx
81  j_begin(itop) = ngllz
82  i_end(itop) = 1
83  j_end(itop) = ngllz
84 
85  i_begin(ileft) = 1
86  j_begin(ileft) = ngllz
87  i_end(ileft) = 1
88  j_end(ileft) = 1
89 
90  ! define i and j points for each edge
91  do ipoin1d = 1,ngllx
92 
93  ivalue(ipoin1d,ibottom) = ipoin1d
94  ivalue_inverse(ipoin1d,ibottom) = ngllx - ipoin1d + 1
95  jvalue(ipoin1d,ibottom) = 1
96  jvalue_inverse(ipoin1d,ibottom) = 1
97 
98  ivalue(ipoin1d,iright) = ngllx
99  ivalue_inverse(ipoin1d,iright) = ngllx
100  jvalue(ipoin1d,iright) = ipoin1d
101  jvalue_inverse(ipoin1d,iright) = ngllz - ipoin1d + 1
102 
103  ivalue(ipoin1d,itop) = ngllx - ipoin1d + 1
104  ivalue_inverse(ipoin1d,itop) = ipoin1d
105  jvalue(ipoin1d,itop) = ngllz
106  jvalue_inverse(ipoin1d,itop) = ngllz
107 
108  ivalue(ipoin1d,ileft) = 1
109  ivalue_inverse(ipoin1d,ileft) = 1
110  jvalue(ipoin1d,ileft) = ngllz - ipoin1d + 1
111  jvalue_inverse(ipoin1d,ileft) = ipoin1d
112 
113  enddo
114 
115  do inum = 1, num_fluid_solid_edges
116  ispec_acoustic = fluid_solid_acoustic_ispec(inum)
117  ispec_elastic = fluid_solid_elastic_ispec(inum)
118 
119  ! one element must be acoustic and the other must be elastic
120  if (ispec_acoustic /= ispec_elastic .and. .not. ispec_is_elastic(ispec_acoustic) .and. &
121  .not. ispec_is_poroelastic(ispec_acoustic) .and. ispec_is_elastic(ispec_elastic)) then
122 
123  ! loop on the four edges of the two elements
124  do iedge_acoustic = 1,nedges
125  do iedge_elastic = 1,nedges
126  ! store the matching topology if the two edges match in inverse order
127  if (ibool(i_begin(iedge_acoustic),j_begin(iedge_acoustic),ispec_acoustic) == &
128  ibool(i_end(iedge_elastic),j_end(iedge_elastic),ispec_elastic) .and. &
129  ibool(i_end(iedge_acoustic),j_end(iedge_acoustic),ispec_acoustic) == &
130  ibool(i_begin(iedge_elastic),j_begin(iedge_elastic),ispec_elastic)) then
131  fluid_solid_acoustic_iedge(inum) = iedge_acoustic
132  fluid_solid_elastic_iedge(inum) = iedge_elastic
133 ! print *,'edge ',iedge_acoustic,' of acoustic element ',ispec_acoustic, &
134 ! ' is in contact with edge ',iedge_elastic,' of elastic element ',ispec_elastic
135  endif
136 
137  enddo
138  enddo
139 
140  endif
141 
142  enddo
143 
144  ! make sure fluid/solid (elastic) matching has been perfectly detected: check that the grid points
145  ! have the same physical coordinates
146  ! loop on all the coupling edges
147 
148  ! user output
149  if (myrank == 0) then
150  write(imain,*) ' Checking fluid/solid edge topology...'
151  call flush_imain()
152  endif
153 
154  do inum = 1,num_fluid_solid_edges

158  iedge_acoustic = fluid_solid_acoustic_iedge(inum)
159 
160  ! get the corresponding edge of the elastic element
161  ispec_elastic = fluid_solid_elastic_ispec(inum)
162  iedge_elastic = fluid_solid_elastic_iedge(inum)
163 
164  ! implement 1D coupling along the edge
165  do ipoin1d = 1,ngllx
166  ! get point values for the elastic side, which matches our side in the inverse direction
167  i = ivalue_inverse(ipoin1d,iedge_elastic)
168  j = jvalue_inverse(ipoin1d,iedge_elastic)
169  iglob = ibool(i,j,ispec_elastic)
170 
171  ! get point values for the acoustic side
172  i = ivalue(ipoin1d,iedge_acoustic)
173  j = jvalue(ipoin1d,iedge_acoustic)
174  iglob2 = ibool(i,j,ispec_acoustic)
175 
176  ! if distance between the two points is not negligible, there is an error, since it should be zero
177  if (sqrt((coord(1,iglob) - coord(1,iglob2))**2 + (coord(2,iglob) - coord(2,iglob2))**2) > tinyval) &
178  call exit_mpi(myrank,'Error in fluid/solid coupling buffer')
179  enddo
180 
181  enddo
182 
183  ! user output
184  if (myrank == 0) then
185  write(imain,*) ' End of fluid/solid edge detection'
186  write(imain,*)
187  call flush_imain()
188  endif
189 
190  endif
191 
192  ! fluid/solid (poroelastic) edge detection
193  ! the two elements (fluid and solid) forming an edge are already known (computed in meshfem2D),
194  ! the common nodes forming the edge are computed here
195  if (coupled_acoustic_poro) then
196 
197  ! user output
198  if (myrank == 0) then
199  write(imain,*)
200  write(imain,*) 'Mixed acoustic/poroelastic simulation'

203  call flush_imain()
204  endif

207  i_begin(ibottom) = 1
208  j_begin(ibottom) = 1
209  i_end(ibottom) = ngllx
210  j_end(ibottom) = 1
211 
212  i_begin(iright) = ngllx
213  j_begin(iright) = 1
214  i_end(iright) = ngllx
215  j_end(iright) = ngllz
216 
217  i_begin(itop) = ngllx
218  j_begin(itop) = ngllz
219  i_end(itop) = 1
220  j_end(itop) = ngllz
221 
222  i_begin(ileft) = 1
223  j_begin(ileft) = ngllz
224  i_end(ileft) = 1
225  j_end(ileft) = 1
226 
227  ! define i and j points for each edge
228  do ipoin1d = 1,ngllx
229 
230  ivalue(ipoin1d,ibottom) = ipoin1d
231  ivalue_inverse(ipoin1d,ibottom) = ngllx - ipoin1d + 1
232  jvalue(ipoin1d,ibottom) = 1
233  jvalue_inverse(ipoin1d,ibottom) = 1
234

236  ivalue_inverse(ipoin1d,iright) = ngllx
237  jvalue(ipoin1d,iright) = ipoin1d
238  jvalue_inverse(ipoin1d,iright) = ngllz - ipoin1d + 1
239 
240  ivalue(ipoin1d,itop) = ngllx - ipoin1d + 1
241  ivalue_inverse(ipoin1d,itop) = ipoin1d
242  jvalue(ipoin1d,itop) = ngllz
243  jvalue_inverse(ipoin1d,itop) = ngllz
244 
245  ivalue(ipoin1d,ileft) = 1
246  ivalue_inverse(ipoin1d,ileft) = 1
247  jvalue(ipoin1d,ileft) = ngllz - ipoin1d + 1
248  jvalue_inverse(ipoin1d,ileft) = ipoin1d
249 
250  enddo
251 
252  do inum = 1, num_fluid_poro_edges
253  ispec_acoustic = fluid_poro_acoustic_ispec(inum)
254  ispec_poroelastic = fluid_poro_poroelastic_ispec(inum)
255 
256  ! one element must be acoustic and the other must be poroelastic
257  if (ispec_acoustic /= ispec_poroelastic .and. .not. ispec_is_poroelastic(ispec_acoustic) .and. &
258  .not. ispec_is_elastic(ispec_acoustic) .and. ispec_is_poroelastic(ispec_poroelastic)) then
259 
260  ! loop on the four edges of the two elements
261  do iedge_acoustic = 1,nedges
262  do iedge_poroelastic = 1,nedges
263 
264  ! store the matching topology if the two edges match in inverse order
265  if (ibool(i_begin(iedge_acoustic),j_begin(iedge_acoustic),ispec_acoustic) == &
266  ibool(i_end(iedge_poroelastic),j_end(iedge_poroelastic),ispec_poroelastic) .and. &
267  ibool(i_end(iedge_acoustic),j_end(iedge_acoustic),ispec_acoustic) == &
268  ibool(i_begin(iedge_poroelastic),j_begin(iedge_poroelastic),ispec_poroelastic)) then
269  fluid_poro_acoustic_iedge(inum) = iedge_acoustic
270  fluid_poro_poroelastic_iedge(inum) = iedge_poroelastic
271  endif
272 
273  enddo
274  enddo
275 
276  endif
277 
278  enddo
279 
280 
281  ! make sure fluid/solid (poroelastic) matching has been perfectly detected: check that the grid points
282  ! have the same physical coordinates
283  ! loop on all the coupling edges
284 
285  ! user output
286  if (myrank == 0) then
287  write(imain,*) ' Checking fluid/solid (poroelastic) edge topology'
288  call flush_imain()
289  endif
290 
291  do inum = 1,num_fluid_poro_edges
292 
293  ! get the edge of the acoustic element
294  ispec_acoustic = fluid_poro_acoustic_ispec(inum)
295  iedge_acoustic = fluid_poro_acoustic_iedge(inum)
296 
297  ! get the corresponding edge of the poroelastic element
298  ispec_poroelastic = fluid_poro_poroelastic_ispec(inum)
299  iedge_poroelastic = fluid_poro_poroelastic_iedge(inum)
300 
301  ! implement 1D coupling along the edge
302  do ipoin1d = 1,ngllx
303 
304  ! get point values for the poroelastic side, which matches our side in the inverse direction
305  i = ivalue_inverse(ipoin1d,iedge_poroelastic)
306  j = jvalue_inverse(ipoin1d,iedge_poroelastic)
307  iglob = ibool(i,j,ispec_poroelastic)
308 
309  ! get point values for the acoustic side
310  i = ivalue(ipoin1d,iedge_acoustic)
311  j = jvalue(ipoin1d,iedge_acoustic)
312  iglob2 = ibool(i,j,ispec_acoustic)
313 
314  ! if distance between the two points is not negligible, there is an error, since it should be zero
315  if (sqrt((coord(1,iglob) - coord(1,iglob2))**2 + (coord(2,iglob) - coord(2,iglob2))**2) > tinyval) &
316  call exit_mpi(myrank,'Error in fluid/solid (poroelastic) coupling buffer')
317 
318  enddo
319 
320  enddo
321 
322  ! user output
323  if (myrank == 0) then
324  write(imain,*) ' End of fluid/solid (poroelastic) edge detection'
325  write(imain,*)
326  call flush_imain()
327  endif
328 
329  endif
330 
331  ! solid/porous edge detection
332  ! the two elements forming an edge are already known (computed in meshfem2D),
333  ! the common nodes forming the edge are computed here
334  if (coupled_elastic_poro) then
335 
336  ! checks
337  if (attenuation_viscoelastic .or. attenuation_poro_fluid_part) &
338  stop 'Attenuation not supported for mixed elastic/poroelastic simulations'
339  if (time_stepping_scheme == 2 .or. time_stepping_scheme == 3) &
340  stop 'RK and LDDRK time scheme not supported for mixed elastic/poroelastic simulations'
341 
342  if (myrank == 0) then
343  write(imain,*)
344  write(imain,*) 'Mixed elastic/poroelastic simulation'
345  write(imain,*)
346  write(imain,*) ' Beginning of solid/porous edge detection'
347  call flush_imain()
348  endif
349 
350  ! define the edges of a given element
351  i_begin(ibottom) = 1
352  j_begin(ibottom) = 1
353  i_end(ibottom) = ngllx
354  j_end(ibottom) = 1
355 
356  i_begin(iright) = ngllx
357  j_begin(iright) = 1
358  i_end(iright) = ngllx
359  j_end(iright) = ngllz
360 
361  i_begin(itop) = ngllx
362  j_begin(itop) = ngllz
363  i_end(itop) = 1
364  j_end(itop) = ngllz
365 
366  i_begin(ileft) = 1
367  j_begin(ileft) = ngllz
368  i_end(ileft) = 1
369  j_end(ileft) = 1
370 
371  ! define i and j points for each edge
372  do ipoin1d = 1,ngllx
373 
374  ivalue(ipoin1d,ibottom) = ipoin1d
375  ivalue_inverse(ipoin1d,ibottom) = ngllx - ipoin1d + 1
376  jvalue(ipoin1d,ibottom) = 1
377  jvalue_inverse(ipoin1d,ibottom) = 1
378 
379  ivalue(ipoin1d,iright) = ngllx
380  ivalue_inverse(ipoin1d,iright) = ngllx
381  jvalue(ipoin1d,iright) = ipoin1d
382  jvalue_inverse(ipoin1d,iright) = ngllz - ipoin1d + 1
383 
384  ivalue(ipoin1d,itop) = ngllx - ipoin1d + 1
385  ivalue_inverse(ipoin1d,itop) = ipoin1d
386  jvalue(ipoin1d,itop) = ngllz
387  jvalue_inverse(ipoin1d,itop) = ngllz
388 
389  ivalue(ipoin1d,ileft) = 1
390  ivalue_inverse(ipoin1d,ileft) = 1
391  jvalue(ipoin1d,ileft) = ngllz - ipoin1d + 1
392  jvalue_inverse(ipoin1d,ileft) = ipoin1d
393 
394  enddo
395 
396 
397  do inum = 1, num_solid_poro_edges
398  ispec_elastic = solid_poro_elastic_ispec(inum)
399  ispec_poroelastic = solid_poro_poroelastic_ispec(inum)
400 
401  ! one element must be elastic and the other must be poroelastic
402  if (ispec_elastic /= ispec_poroelastic .and. ispec_is_elastic(ispec_elastic) .and. &
403  ispec_is_poroelastic(ispec_poroelastic)) then
404 
405  ! loop on the four edges of the two elements
406  do iedge_poroelastic = 1,nedges
407  do iedge_elastic = 1,nedges
408 
409  ! store the matching topology if the two edges match in inverse order
410  if (ibool(i_begin(iedge_poroelastic),j_begin(iedge_poroelastic),ispec_poroelastic) == &
411  ibool(i_end(iedge_elastic),j_end(iedge_elastic),ispec_elastic) .and. &
412  ibool(i_end(iedge_poroelastic),j_end(iedge_poroelastic),ispec_poroelastic) == &
413  ibool(i_begin(iedge_elastic),j_begin(iedge_elastic),ispec_elastic)) then
414  solid_poro_elastic_iedge(inum) = iedge_elastic
415  solid_poro_poroelastic_iedge(inum) = iedge_poroelastic
416  endif
417 
418  enddo
419  enddo
420 
421  endif
422 
423  enddo
424 
425  ! make sure solid/porous matching has been perfectly detected: check that the grid points
426  ! have the same physical coordinates
427  ! loop on all the coupling edges
428 
429  if (myrank == 0) then
430  write(imain,*) ' Checking solid/porous edge topology...'
431  call flush_imain()
432  endif
433 
434  do inum = 1,num_solid_poro_edges
435 
436  ! get the edge of the elastic element
437  ispec_elastic = solid_poro_elastic_ispec(inum)
438  iedge_elastic = solid_poro_elastic_iedge(inum)
439 
440  ! get the corresponding edge of the poroelastic element
441  ispec_poroelastic = solid_poro_poroelastic_ispec(inum)
442  iedge_poroelastic = solid_poro_poroelastic_iedge(inum)
443 
444  ! implement 1D coupling along the edge
445  do ipoin1d = 1,ngllx
446 
447  ! get point values for the poroelastic side, which matches our side in the inverse direction
448  i = ivalue_inverse(ipoin1d,iedge_elastic)
449  j = jvalue_inverse(ipoin1d,iedge_elastic)

451 
452  ! get point values for the elastic side

459  call exit_mpi(myrank,'Error in solid/porous coupling buffer')
460 
461  enddo
462 
463  enddo
464 
465  if (myrank == 0) then
466  write(imain,*) ' End of solid/porous edge detection'
467  write(imain,*)
468  call flush_imain()
469  endif
470 
471  endif
472 
473  ! synchronizes all processes
474  call synchronize_all()
475 
476  ! exclude common points between acoustic absorbing edges and acoustic/elastic matching interfaces
477  if (coupled_acoustic_elastic .and. anyabs) then
478 
479  if (myrank == 0) then
480  write(imain,*) 'excluding common points between acoustic absorbing edges '// &
481  'and acoustic/elastic matching interfaces, if any'
482  call flush_imain()
483  endif
484  ! excludes common points in acoustic domain
485  call get_coupling_edges_exclude_common_points(nelemabs,numabs,num_fluid_solid_edges, &

487  ibegin_edge1,ibegin_edge2,ibegin_edge3,ibegin_edge4, &

490

495  write(imain,*) 'excluding common points between acoustic absorbing edges'// &

498  endif

527  subroutine get_coupling_edges_exclude_common_points(nelemabs,numabs,num_domain_edges, &
528  domain_ispec,domain_iedge, &
529  ibegin_edge1,ibegin_edge2,ibegin_edge3,ibegin_edge4, &
530  iend_edge1,iend_edge2,iend_edge3,iend_edge4)
531 
532 ! excludes common GLL points (in one of the domains) between coupling domains to correct absorbing boundary
533 
534  use constants, only: ibottom,itop,ileft,iright,ngllx,ngllz
535 
536  implicit none
537 
538  integer,intent(in) :: nelemabs
539  integer,dimension(nelemabs),intent(in) ::numabs
540 
541  integer,intent(in) :: num_domain_edges
542  integer,dimension(num_domain_edges),intent(in) :: domain_ispec,domain_iedge
543 
544  integer,dimension(nelemabs),intent(inout) :: ibegin_edge1,ibegin_edge2,ibegin_edge3,ibegin_edge4
545  integer,dimension(nelemabs),intent(inout) :: iend_edge1,iend_edge2,iend_edge3,iend_edge4
546 
547  ! local parameters
548  integer :: ispec,ispecabs,inum
549  integer :: ispec_domain,iedge_domain
550 
551  ! loop on all the absorbing elements
552  do ispecabs = 1,nelemabs
553 
554  ispec = numabs(ispecabs)
555 
556  ! loop on all the coupling edges
557  do inum = 1,num_domain_edges
558 
559  ! get the edge of the acoustic element
560  ispec_domain = domain_ispec(inum)
561  iedge_domain = domain_iedge(inum)
562 
563  ! if acoustic absorbing element and acoustic/poroelastic coupled element is the same
564  if (ispec_domain == ispec) then
565 
566  if (iedge_domain == ibottom) then
567  ibegin_edge4(ispecabs) = 2
568  ibegin_edge2(ispecabs) = 2
569  endif
570 
571  if (iedge_domain == itop) then
572  iend_edge4(ispecabs) = ngllz - 1
573  iend_edge2(ispecabs) = ngllz - 1
574  endif
575 
576  if (iedge_domain == ileft) then
577  ibegin_edge1(ispecabs) = 2
578  ibegin_edge3(ispecabs) = 2
579  endif
