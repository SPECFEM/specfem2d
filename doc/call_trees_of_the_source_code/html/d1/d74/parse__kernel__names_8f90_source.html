34  subroutine parse_kernel_names(kernel_names_comma_delimited,kernel_names,nker)
35 
36  use postprocess_par, only: max_string_len, max_kernel_names
37 
38  implicit none
39 
40  character(len=MAX_STRING_LEN),intent(in) :: kernel_names_comma_delimited
41  character(len=MAX_STRING_LEN),intent(inout) :: kernel_names(max_kernel_names)
42  integer,intent(out) :: nker
43 
44  ! local parameters
45  integer :: iker
46  character(len=MAX_STRING_LEN) :: tmp
47  character,parameter :: delimiter = ','
48 
49  ! gets first name/token
50  iker = 1
51  call strtok(kernel_names_comma_delimited, delimiter, kernel_names(iker))
52 
53  ! null-string as first argument for successive strtok-calls
54  tmp(1:1) = char(0)
55 
56  ! gets next names/tokens (strtok will return null-terminated token when finished)
57  do while (kernel_names(iker)(1:1) /= char(0))
58  ! increases name/token number
59  iker = iker + 1
60  ! gets next successive token (with null-terminated string as first argument)
61  call strtok(tmp, delimiter, kernel_names(iker))
62  enddo
63 
64  ! number of kernel names
65  nker = iker-1
66 
67  ! checks name lengths (e.g. if kernel_name argument is "vsv,vsh," we will have a 3. kernel name with empty string)
68  do iker = 1,nker
69  if (len_trim(kernel_names(iker)) == 0) then
70  print *,'Error encountered kernel name with zero length: kernel name number ',iker,' out of ',nker,' is empty'
71  print *,'Please check your kernel_names argument...'
72  stop 'Error kernel name with zero length'
73  endif
74  enddo
75 
76  end subroutine parse_kernel_names

85  subroutine strtok (source_string, delimiter, token)
86 
87 ! @(#) Tokenize a string in a similar manner to C routine strtok(3c).
88 !
89 ! Usage: First call STRTOK() with the string to tokenize as SOURCE_STRING,
90 ! and the delimiter list used to tokenize SOURCE_STRING in delimiter.
91 !
92 ! then, if the returned value is not equal to char(0), keep calling until it is
93 ! with SOURCE_STRING set to char(0).
94 !
95 ! STRTOK will return a token on each call until the entire line is processed,
96 ! which it signals by returning char(0).
97 !
98 ! Input: source_string = Source string to tokenize.
99 ! delimiter = delimiter string. Used to determine the beginning/end of each token in a string.
100 !
101 ! Output: strtok()
102 !
103 ! LIMITATIONS:
104 ! can not be called with a different string until current string is totally processed, even from different procedures
105 
106  use postprocess_par, only: max_string_len
107 
108  ! PARAMETERS:
109  character(len=MAX_STRING_LEN), intent(in) :: source_string
110  character(len=1), intent(in) :: delimiter
111  character(len=MAX_STRING_LEN), intent(out) :: token
112 
113  ! SAVED VALUES:
114  character(len=MAX_STRING_LEN),save :: saved_string
115  integer,save :: isaved_start ! points to beginning of unprocessed data
116  integer,save :: isource_len ! length of original input string
117 
118  ! LOCAL VALUES:
119  integer :: ibegin ! beginning of token to return
120  integer :: ifinish ! end of token to return
121 
122  ! initialize stored copy of input string and pointer into input string on first call
123  if (source_string(1:1) /= char(0)) then
124  isaved_start = 1 ! beginning of unprocessed data
125  saved_string = source_string ! save input string from first call in series
126  isource_len = len(saved_string) ! length of input string from first call
127  endif
128 
129  token = ''
130  ibegin = isaved_start
131 
132  ! sets first index ibegin to beginning of (next) token
133  do while (.true.)
134  if ((ibegin <= isource_len) .and. (index(delimiter,saved_string(ibegin:ibegin)) /= 0)) then
135  ! delimiter is encountered, starts with next index (next token)
136  ibegin = ibegin + 1
137  else
138  ! exits do-loop
139  exit
140  endif
141  enddo
142 
143  if (ibegin > isource_len) then
144  token = char(0)
145  return
146  endif
147 
148  ! sets second index ifinish to end of token (including delimiter)
149  ifinish = ibegin
150 
151  do while (.true.)
152  if ((ifinish <= isource_len) .and. (index(delimiter,saved_string(ifinish:ifinish)) == 0)) then
153  ! delimiter is not encountered yet, increases finish index
154  ifinish = ifinish + 1
155  else
156  ! exits do-loop
157  exit
158  endif
159  enddo
160 
161  ! sets token string
162  !strtok = "["//saved_string(ibegin:ifinish-1)//"]"
163  token = saved_string(ibegin:ifinish-1)
164  isaved_start = ifinish
165 
166  end subroutine strtok
