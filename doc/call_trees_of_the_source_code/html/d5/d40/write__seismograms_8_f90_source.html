34 ! write seismograms to text files
35 
36  subroutine write_seismograms()
37 
38  use constants, only: zero
39  use specfem_par
40  use specfem_par_gpu, only: mesh_pointer
41 
42  implicit none
43 
44  ! local parameters
45  integer :: i, j, iglob, irecloc, ispec
46  double precision :: valux,valuz,valcurl
47 
48  ! vector field in an element
49  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLZ) :: vector_field_element
50  ! pressure in an element
51  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: pressure_element
52  ! curl in an element
53  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLZ) :: curl_element
54 
55  ! checks subsampling recurrence
56  if (mod(it-1,subsamp_seismos) == 0) then
57 
58  ! update position in seismograms
59  seismo_current = seismo_current + 1
60 
61  ! check for edge effects
62  if (seismo_current < 1 .or. seismo_current > nstep_between_output_seismos/subsamp_seismos) &
63  stop 'Error: seismo_current out of bounds in recording of seismograms'
64 
65  ! updates local receiver records
66  if (nrecloc > 0) then
67 
68  ! computes seismogram entry for all local receivers
69  if (.not. gpu_mode) then
70  ! on CPU
71  do irecloc = 1,nrecloc
72 
73  ispec = ispec_selected_rec_loc(irecloc)
74 
75  !debug
76  !print *,'ispec',ispec,ispec_is_acoustic(ispec),ispec_is_elastic(ispec), &
77  ! ispec_is_gravitoacoustic(ispec),ispec_is_poroelastic(ispec)
78 
79  ! initializes local element arrays
80  vector_field_element(:,:,:) = 0._custom_real
81  pressure_element(:,:) = 0._custom_real
82  curl_element(:,:) = 0._custom_real
83 
84  ! compute seismo/pressure/curl in this element
85  select case (seismotype)
86  case (1)
87  ! displacement
88  call compute_vector_one_element(potential_acoustic,potential_gravitoacoustic, &
89  potential_gravito,displ_elastic,displs_poroelastic, &
90  ispec,vector_field_element)
91  case (2)
92  ! velocity
93  call compute_vector_one_element(potential_dot_acoustic,potential_dot_gravitoacoustic, &
94  potential_dot_gravito,veloc_elastic,velocs_poroelastic, &
95  ispec,vector_field_element)
96  case (3)
97  ! acceleration
98  call compute_vector_one_element(potential_dot_dot_acoustic,potential_dot_dot_gravitoacoustic, &
99  potential_dot_dot_gravito,accel_elastic,accels_poroelastic, &
100  ispec,vector_field_element)
101  case (4)
102  ! pressure
103  call compute_pressure_one_element(ispec,pressure_element)
104 
105  case (5)
106  ! displacement
107  call compute_vector_one_element(potential_acoustic,potential_gravitoacoustic, &
108  potential_gravito,displ_elastic,displs_poroelastic, &
109  ispec,vector_field_element)
110  ! curl of displacement
111  call compute_curl_one_element(ispec,curl_element)
112 
113  case (6)
114  ! fluid potential
115  ! uses pressure_element to store local element values
116  if (ispec_is_acoustic(ispec)) then
117  do j = 1,ngllz
118  do i = 1,ngllx
119  iglob = ibool(i,j,ispec)
120  pressure_element(i,j) = potential_acoustic(iglob)
121  enddo
122  enddo
123  else if (ispec_is_gravitoacoustic(ispec)) then
124  do j = 1,ngllz
125  do i = 1,ngllx
126  iglob = ibool(i,j,ispec)
127  pressure_element(i,j) = potential_gravitoacoustic(iglob)
128  enddo
129  enddo
130  endif
131 
132  case default
133  stop 'Invalid seismotype for writing seismograms'
134  end select
135 
136  ! perform the general interpolation using Lagrange polynomials
137  call compute_interpolated_dva(irecloc,ispec,vector_field_element,pressure_element,curl_element, &
138  valux,valuz,valcurl)
139 
140  ! rotate seismogram components if needed, except if recording pressure, which is a scalar
141  if (seismotype == 4 .or. seismotype == 6) then
142  ! pressure/potential type has only single component
143  sisux(seismo_current,irecloc) = valux
144  sisuz(seismo_current,irecloc) = zero
145  else
146  if (p_sv) then
147  sisux(seismo_current,irecloc) = cosrot_irec(irecloc)*valux + sinrot_irec(irecloc)*valuz
148  sisuz(seismo_current,irecloc) = - sinrot_irec(irecloc)*valux + cosrot_irec(irecloc)*valuz
149  else
150  sisux(seismo_current,irecloc) = valux
151  sisuz(seismo_current,irecloc) = zero
152  endif
153  ! additional curl case
154  if (seismotype == 5) siscurl(seismo_current,irecloc) = valcurl
155  endif
156  enddo ! irecloc
157 
158  else
159  ! on GPU
160  if (simulation_type == 1) then
161  ! Simulating seismograms
162  if (use_trick_for_better_pressure) then
163  call compute_seismograms_cuda(mesh_pointer,seismotype,sisux,sisuz,seismo_current, &
164  nstep_between_output_seismos/subsamp_seismos, &
165  elastic_simulation,acoustic_simulation,1)
166  else
167  call compute_seismograms_cuda(mesh_pointer,seismotype,sisux,sisuz,seismo_current, &
168  nstep_between_output_seismos/subsamp_seismos, &
169  elastic_simulation,acoustic_simulation,0)
170  endif
171  ! note: curl not implemented yet
172  endif
173  endif ! GPU_MODE
174 
175  endif ! nrecloc
176 
177  endif ! subsamp_seismos
178 
179  ! save temporary or final seismograms
180  ! suppress seismograms if we generate traces of the run for analysis with "ParaVer", because time consuming
181  if (mod(it,nstep_between_output_seismos) == 0 .or. it == nstep) then
182  call write_seismograms_to_file(sisux,sisuz,siscurl)
183 
184  ! updates current seismogram offsets
185  seismo_offset = seismo_offset + seismo_current
186  seismo_current = 0
187  endif
188 
189  end subroutine write_seismograms
190 
191 !================================================================
192 
193  subroutine write_seismograms_to_file(sisux,sisuz,siscurl)
194 
195 #ifdef USE_MPI
196  use mpi
197 #endif
198 
199  use constants, only: ndim,max_length_network_name,max_length_station_name,output_files
200 
201  use specfem_par, only: station_name,network_name,nstep,islice_selected_rec,nrec,myrank,deltat,seismotype,t0, &
202  nstep_between_output_seismos,subsamp_seismos,nrecloc, &
203  seismo_offset,seismo_current,p_sv,su_format,save_ascii_seismograms, &
204  save_binary_seismograms_single,save_binary_seismograms_double,x_source,z_source
205 
206  implicit none
207 
208  double precision,dimension(NSTEP_BETWEEN_OUTPUT_SEISMOS/subsamp_seismos,nrecloc),intent(in) :: sisux,sisuz,siscurl
209 
210  ! local parameters
211  logical :: save_binary_seismograms
212  integer :: irec,length_station_name,length_network_name,iorientation,isample,number_of_components
213 
214  character(len=4) :: channel
215  character(len=1) :: component
216  character(len=4) :: suffix
217  character(len=150) :: sisname
218 
219  ! to write seismograms in single precision SEP and double precision binary format
220  double precision, dimension(:,:,:), allocatable :: buffer_binary
221  double precision :: time_t
222 
223  integer :: irecloc
224  integer :: ier
225 
226 ! write seismograms
227 
228  ! save displacement, velocity, acceleration or pressure
229  if (seismotype == 1) then
230  component = 'd'
231  else if (seismotype == 2) then
232  component = 'v'
233  else if (seismotype == 3) then
234  component = 'a'
235  else if (seismotype == 4 .or. seismotype == 6) then
236  component = 'p'
237  else if (seismotype == 5) then
238  component = 'c'
239  else
240  call exit_mpi(myrank,'wrong component to save for seismograms')
241  endif
242 
243  ! only one seismogram if pressures or SH (membrane) waves
244  if (seismotype == 4 .or. seismotype == 6 .or. .not. p_sv) then
245  number_of_components = 1
246  else if (seismotype == 5) then
247  ! adds curl
248  number_of_components = ndim + 1
249  else
250  number_of_components = ndim
251  endif
252 
253  allocate(buffer_binary(nstep_between_output_seismos/subsamp_seismos,nrec,number_of_components),stat=ier)
254  if (ier /= 0) stop 'Error allocating array buffer_binary'
255  buffer_binary(:,:,:) = 0.d0
256 
257  ! see if we need to save any seismogram in binary format
258  save_binary_seismograms = save_binary_seismograms_single .or. save_binary_seismograms_double
259 
260  ! binary file output
261  if (save_binary_seismograms .and. myrank == 0) then
262  ! filename extension
263  if (.not. su_format) then
264  suffix = '.bin'
265  else
266  suffix = '.su'
267  endif
268 
269  ! deletes old files
270  if (seismo_offset == 0) then
271  open(unit=12,file=trim(output_files)//'Ux_file_single'//suffix,status='unknown')
272  close(12,status='delete')
273 
274  open(unit=12,file=trim(output_files)//'Ux_file_double'//suffix,status='unknown')
275  close(12,status='delete')
276 
277  open(unit=12,file=trim(output_files)//'Uy_file_single'//suffix,status='unknown')
278  close(12,status='delete')
279 
280  open(unit=12,file=trim(output_files)//'Uy_file_double'//suffix,status='unknown')
281  close(12,status='delete')
282 
283  open(unit=12,file=trim(output_files)//'Uz_file_single'//suffix,status='unknown')
284  close(12,status='delete')
285 
286  open(unit=12,file=trim(output_files)//'Uz_file_double'//suffix,status='unknown')
287  close(12,status='delete')
288 
289  open(unit=12,file=trim(output_files)//'Up_file_single'//suffix,status='unknown')
290  close(12,status='delete')
291 
292  open(unit=12,file=trim(output_files)//'Up_file_double'//suffix,status='unknown')
293  close(12,status='delete')
294 
295  open(unit=12,file=trim(output_files)//'Uc_file_single'//suffix,status='unknown')
296  close(12,status='delete')
297 
298  open(unit=12,file=trim(output_files)//'Uc_file_double'//suffix,status='unknown')
299  close(12,status='delete')
300  endif
301 
302  ! write the new files
303  if (save_binary_seismograms_single) then
304  if (seismotype == 4 .or. seismotype == 6) then
305  open(unit=12,file=trim(output_files)//'Up_file_single'//suffix,status='unknown',access='direct',recl=4)
306  else if (.not. p_sv) then
307  open(unit=12,file=trim(output_files)//'Uy_file_single'//suffix,status='unknown',access='direct',recl=4)
308  else
309  open(unit=12,file=trim(output_files)//'Ux_file_single'//suffix,status='unknown',access='direct',recl=4)
310  endif
311  endif
312 
313  if (save_binary_seismograms_double) then
314  if (seismotype == 4 .or. seismotype == 6) then
315  ! continue without output
316  else if (.not. p_sv) then
317  open(unit=13,file=trim(output_files)//'Uz_file_double'//suffix,status='unknown',access='direct',recl=8)
318  else
319  open(unit=13,file=trim(output_files)//'Ux_file_double'//suffix,status='unknown',access='direct',recl=8)
320  endif
321  endif
322 
323  ! no Z component seismogram if pressure
324  if (seismotype /= 4 .and. seismotype /= 6 .and. p_sv) then
325  if (save_binary_seismograms_single) &
326  open(unit=14,file=trim(output_files)//'Uz_file_single'//suffix,status='unknown',access='direct',recl=4)
327  if (save_binary_seismograms_double) &
328  open(unit=15,file=trim(output_files)//'Uz_file_double'//suffix,status='unknown',access='direct',recl=8)
329  endif
330 
331  ! curl output
332  if (seismotype == 5) then
333  if (save_binary_seismograms_single) &
334  open(unit=16,file=trim(output_files)//'Uc_file_single'//suffix,status='unknown',access='direct',recl=4)
335  if (save_binary_seismograms_double) &
336  open(unit=17,file=trim(output_files)//'Uc_file_double'//suffix,status='unknown',access='direct',recl=8)
337  endif
338  endif ! save_binary_seismograms
339 
340  ! sets seismogram values
341  irecloc = 0
342  do irec = 1,nrec
343  if (myrank == 0) then
344 
345  if (myrank == islice_selected_rec(irec)) then
346  irecloc = irecloc + 1
347 
348  ! fills buffer
349  buffer_binary(:,irec,1) = sisux(:,irecloc)
350  if (number_of_components == 2) then
351  buffer_binary(:,irec,2) = sisuz(:,irecloc)
352  else if (number_of_components == 3) then
353  ! adds curl trace
354  buffer_binary(:,irec,2) = sisuz(:,irecloc)
355  buffer_binary(:,irec,3) = siscurl(:,irecloc)
356  endif
357 
358 #ifdef USE_MPI
359  else
360  ! collects seismogram components on master
361  call recv_dp(buffer_binary(1,irec,1), nstep_between_output_seismos/subsamp_seismos, islice_selected_rec(irec), irec)
362 
363  if (number_of_components == 2) then
364  call recv_dp(buffer_binary(1,irec,2), nstep_between_output_seismos/subsamp_seismos, islice_selected_rec(irec), irec)
365  endif
366 
367  if (number_of_components == 3) then
368  call recv_dp(buffer_binary(1,irec,2), nstep_between_output_seismos/subsamp_seismos, islice_selected_rec(irec), irec)
369  call recv_dp(buffer_binary(1,irec,3), nstep_between_output_seismos/subsamp_seismos, islice_selected_rec(irec), irec)
370  endif
371 #endif
372  endif
373  endif ! myrank == 0
374  enddo
375 
376  irecloc = 0
377  do irec = 1,nrec
378  if (myrank == 0) then
379 
380  if (.not. su_format) then
381 
382  if (save_ascii_seismograms) then
383 
384  ! write trace
385  do iorientation = 1,number_of_components
386 
387  if (iorientation == 1) then
388  channel = 'BXX'
389  else if (iorientation == 2) then
390  channel = 'BXZ'
391  else if (iorientation == 3) then
392  channel = 'cur'
393  else
394  call exit_mpi(myrank,'incorrect channel value')
395  endif
396 
397  ! in case of pressure, use different abbreviation
398  if (seismotype == 4 .or. seismotype == 6) channel = 'PRE'
399 
400  ! in case of SH (membrane) waves, use different abbreviation
401  if (.not. p_sv) channel = 'BXY'
402 
403  ! create the name of the seismogram file for each slice
404  ! file name includes the name of the station, the network and the component
405  length_station_name = len_trim(station_name(irec))
406  length_network_name = len_trim(network_name(irec))
407 
408  ! check that length conforms to standard
409  if (length_station_name < 1 .or. length_station_name > max_length_station_name) then
410  call exit_mpi(myrank,'wrong length of station name')
411  endif
412  if (length_network_name < 1 .or. length_network_name > max_length_network_name) then
413  call exit_mpi(myrank,'wrong length of network name')
414  endif
415 
416  write(sisname,"(a,a,'.',a,'.',a3,'.sem',a1)") &
417  trim(output_files),network_name(irec)(1:length_network_name),station_name(irec)(1:length_station_name),channel,component
418 
419  ! deletes old seismogram file when starting to write output
420  if (seismo_offset == 0) then
421  open(unit=11,file=sisname(1:len_trim(sisname)),status='unknown')
422  close(11,status='delete')
423  endif
424 
425  open(unit=11,file=sisname(1:len_trim(sisname)),status='unknown',position='append')
426 
427  ! make sure we never write more than the maximum number of time steps
428  ! subtract offset of the source to make sure travel time is correct
429  do isample = 1,seismo_current
430 
431  ! forward time
432  time_t = dble(seismo_offset + isample - 1) * deltat - t0
433 
434  write(11,*) time_t,' ',buffer_binary(isample,irec,iorientation)
435  enddo
436 
437  close(11)
438  enddo ! iorientation
439 
440  endif ! save_ASCII_seismograms
441 
442  ! write binary seismogram
443  if (save_binary_seismograms) then
444  do isample = 1, seismo_current
445  if (save_binary_seismograms_single) &
446  write(12,rec=(irec-1)*nstep+seismo_offset+isample) sngl(buffer_binary(isample,irec,1))
447  if (save_binary_seismograms_double) &
448  write(13,rec=(irec-1)*nstep+seismo_offset+isample) buffer_binary(isample,irec,1)
449 
450  if (seismotype /= 4 .and. seismotype /= 6 .and. p_sv) then
451  if (save_binary_seismograms_single) &
452  write(14,rec=(irec-1)*nstep+seismo_offset+isample) sngl(buffer_binary(isample,irec,2))
453  if (save_binary_seismograms_double) &
454  write(15,rec=(irec-1)*nstep+seismo_offset+isample) buffer_binary(isample,irec,2)
455  endif
456 
457  if (seismotype == 5) then
458  if (save_binary_seismograms_single) &
459  write(16,rec=(irec-1)*nstep+seismo_offset+isample) sngl(buffer_binary(isample,irec,3))
460  if (save_binary_seismograms_double) &
461  write(17,rec=(irec-1)*nstep+seismo_offset+isample) buffer_binary(isample,irec,3)
462  endif
463  enddo
464  endif
465 
466  else
467  ! if SU_FORMAT
468  call write_output_su(x_source(1),z_source(1),irec,buffer_binary,number_of_components)
469  endif
470 
471 #ifdef USE_MPI
472  else
473  ! slave processes (myrank > 0)
474  ! sends seismogram values to master
475  if (myrank == islice_selected_rec(irec)) then
476  irecloc = irecloc + 1
477  call send_dp(sisux(1,irecloc), nstep_between_output_seismos/subsamp_seismos, 0, irec)
478 
479  if (number_of_components >= 2) then
480  call send_dp(sisuz(1,irecloc), nstep_between_output_seismos/subsamp_seismos, 0, irec)
481  endif
482 
483  if (number_of_components == 3) then
484  call send_dp(siscurl(1,irecloc), nstep_between_output_seismos/subsamp_seismos, 0, irec)
485  endif
486 
487  endif
488 #endif
489 
490  endif ! myrank
491 
492  enddo
493 
494  ! closes files
495  if (save_binary_seismograms_single) close(12)
496  if (save_binary_seismograms_double) close(13)
497  if (seismotype /= 4 .and. seismotype /= 6 .and. p_sv) then
498  if (save_binary_seismograms_single) close(14)
499  if (save_binary_seismograms_double) close(15)
500  endif
501  if (seismotype == 5) then
502  if (save_binary_seismograms_single) close(16)
503  if (save_binary_seismograms_double) close(17)
504  endif
505 
506  ! frees temporary buffer
507  deallocate(buffer_binary)
508 
509  end subroutine write_seismograms_to_file
